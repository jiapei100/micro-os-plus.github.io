82  return_t
83  Usbh_wrapper::do_power (Power state) noexcept
84  {
85  assert(driver_ != nullptr);
86 
87  return_t status;
88 
89  if (state == Power::full)
90  {
91  status = driver_->Initialize (c_cb_port_func_, c_cb_pipe_func_);
92  if (status != ARM_DRIVER_OK)
93  {
94  return status;
95  }
96  }
97 
98  status = driver_->PowerControl ((ARM_POWER_STATE) state);
99 
100  if (state == Power::off)
101  {
102  driver_->Uninitialize ();
103  }
104 
105  return status;
106  }
