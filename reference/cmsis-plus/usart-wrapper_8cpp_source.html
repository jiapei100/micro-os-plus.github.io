88  return_t
89  Usart_wrapper::do_power (Power state) noexcept
90  {
91  return_t status;
92 
93  if (state == Power::full)
94  {
95  status = driver_->Initialize (c_cb_func_);
96  if (status != ARM_DRIVER_OK)
97  {
98  return status;
99  }
100  }
101 
102  status = driver_->PowerControl ((ARM_POWER_STATE) state);
103 
104  if (state == Power::off)
105  {
106  driver_->Uninitialize ();
107  }
108 
109  return status;
