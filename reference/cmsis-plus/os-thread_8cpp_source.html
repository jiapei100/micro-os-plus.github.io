854  result_t
855  Thread::_try_wait (thread::sigset_t mask, thread::sigset_t* oflags,
856  flags::mode_t mode)
857  {
858  if ((mask != 0) && ((mode & flags::mode::all) != 0))
859  {
860  // Only if all desires signals are raised we're done.
861  if ((sig_mask_ & mask) == mask)
862  {
863  if (oflags != nullptr)
864  {
865  *oflags = sig_mask_;
866  }
867  // Clear desired signals.
868  sig_mask_ &= ~mask;
869  return result::ok;
870  }
871  }
872  else if ((mask == 0) || ((mode & flags::mode::any) != 0))
873  {
874  // Any signal will do it.
875  if (sig_mask_ != 0)
876  {
877  // Possibly return .
878  if (oflags != nullptr)
879  {
880  *oflags = sig_mask_;
881  }
882  // Since we returned them all, also clear them all.
883  sig_mask_ = 0;
884  return result::ok;
885  }
886  }
887 
888  return EAGAIN;
