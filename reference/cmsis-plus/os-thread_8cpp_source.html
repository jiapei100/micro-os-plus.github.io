155  void
156  thread::stack::initialize (void)
157  {
158  // Align the bottom of the stack.
159  void* pa = bottom_address_;
160  bottom_address_ = static_cast<stack::element_t*> (std::align (
161  sizeof(stack::allocation_element_t), stack::min_size (), pa,
162  size_bytes_));
163 
164  // If there is not enough space for the minimal stack, fail.
165  os_assert_throw(bottom_address_ != nullptr, ENOMEM);
166 
167  element_t* p = bottom_address_;
168  element_t* pend = top ();
169 
170  // Initialise the entire stack with the magic word.
171  for (; p < pend; ++p)
172  {
173  *p = magic;
174  }
175 
176  // Compute the actual size. The -1 is to leave space for the magic.
177  size_bytes_ = ((static_cast<std::size_t> (p - bottom_address_) - 1)
178  * sizeof(element_t));
179  }
