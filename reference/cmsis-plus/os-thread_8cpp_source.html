843  result_t
844  Thread::_try_wait (thread::sigset_t mask, thread::sigset_t* oflags,
845  flags::mode_t mode)
846  {
847  if ((mask != 0) && ((mode & flags::mode::all) != 0))
848  {
849  // Only if all desires signals are raised we're done.
850  if ((sig_mask_ & mask) == mask)
851  {
852  if (oflags != nullptr)
853  {
854  *oflags = sig_mask_;
855  }
856  // Clear desired signals.
857  sig_mask_ &= ~mask;
858  return result::ok;
859  }
860  }
861  else if ((mask == 0) || ((mode & flags::mode::any) != 0))
862  {
863  // Any signal will do it.
864  if (sig_mask_ != 0)
865  {
866  // Possibly return .
867  if (oflags != nullptr)
868  {
869  *oflags = sig_mask_;
870  }
871  // Since we returned them all, also clear them all.
872  sig_mask_ = 0;
873  return result::ok;
874  }
875  }
876 
877  return EAGAIN;
878  }
