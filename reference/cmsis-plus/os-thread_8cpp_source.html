877  result_t
878  Thread::_try_wait (thread::sigset_t mask, thread::sigset_t* oflags,
879  flags::mode_t mode)
880  {
881  if ((mask != 0) && ((mode & flags::mode::all) != 0))
882  {
883  // Only if all desires signals are raised we're done.
884  if ((sig_mask_ & mask) == mask)
885  {
886  if (oflags != nullptr)
887  {
888  *oflags = sig_mask_;
889  }
890  // Clear desired signals.
891  sig_mask_ &= ~mask;
892  return result::ok;
893  }
894  }
895  else if ((mask == 0) || ((mode & flags::mode::any) != 0))
896  {
897  // Any signal will do it.
898  if (sig_mask_ != 0)
899  {
900  // Possibly return .
901  if (oflags != nullptr)
902  {
903  *oflags = sig_mask_;
904  }
905  // Since we returned them all, also clear them all.
906  sig_mask_ = 0;
907  return result::ok;
908  }
909  }
910 
911  return EWOULDBLOCK;
912  }
