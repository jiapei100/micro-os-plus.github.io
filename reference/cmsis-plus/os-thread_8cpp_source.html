822  Thread::_try_wait (thread::sigset_t mask, thread::sigset_t* oflags,
823  flags::mode_t mode)
824  {
825  if ((mask != 0) && ((mode & flags::mode::all) != 0))
826  {
827  // Only if all desires signals are raised we're done.
828  if ((sig_mask_ & mask) == mask)
829  {
830  if (oflags != nullptr)
831  {
832  *oflags = sig_mask_;
833  }
834  // Clear desired signals.
835  sig_mask_ &= ~mask;
836  return result::ok;
837  }
838  }
839  else if ((mask == 0) || ((mode & flags::mode::any) != 0))
840  {
841  // Any signal will do it.
842  if (sig_mask_ != 0)
843  {
844  // Possibly return .
845  if (oflags != nullptr)
846  {
847  *oflags = sig_mask_;
848  }
849  // Since we returned them all, also clear them all.
850  sig_mask_ = 0;
851  return result::ok;
852  }
853  }
