332  thread::_construct (func_t function, func_args_t args,
333  const attributes& attr, void* stack_address,
334  std::size_t stack_size_bytes)
335  {
336  os_assert_throw(!scheduler::in_handler_mode (), EPERM);
337 
338  assert(function != nullptr);
339  assert(attr.th_priority != priority::none);
340 
341  clock_ = attr.clock != nullptr ? attr.clock : &sysclock;
342 
343  if (stack_address != nullptr)
344  {
345  // The attributes should not define any storage in this case.
346  if (attr.th_stack_size_bytes > stack::min_size ())
347  {
348  assert(attr.th_stack_address == nullptr);
349  }
350 
351  context_.stack_.size_bytes_ = stack_size_bytes;
352  context_.stack_.bottom_address_ =
353  static_cast<stack::element_t*> (stack_address);
354  }
355  else
356  {
357  context_.stack_.size_bytes_ = attr.th_stack_size_bytes;
358  context_.stack_.bottom_address_ =
359  static_cast<stack::element_t*> (attr.th_stack_address);
360  }
361 
362 #if defined(OS_TRACE_RTOS_THREAD)
363  trace::printf ("%s @%p %s p%d %d\n", __func__, this, name (),
364  attr.th_priority, context_.stack_.size_bytes_);
365 #endif
366 
367  {
368  // Prevent the new thread to execute before all members are set.
369  scheduler::critical_section cs; // ----- Critical section -----
370 
371  // Get attributes from user structure.
372  prio_ = attr.th_priority;
373 
374  func_ = function;
375  func_args_ = args;
376 
377  parent_ = this_thread::_thread ();
378  if (scheduler::started () && (parent_ != nullptr))
379  {
380  parent_->children_.link (*this);
381  }
382  else
383  {
384  scheduler::top_threads_list_.link (*this);
385  }
386

388 
389  port::thread::create (this);
390  sched_state_ = state::ready;

392 #else

395  void* p = context_.stack_.bottom_address_;
396  context_.stack_.bottom_address_ =
397  static_cast<stack::element_t*> (std::align (
398  sizeof(stack::element_t), stack::min_size (), p,
