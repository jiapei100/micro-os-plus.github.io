214  void
215  newlib_nano::do_deallocate (void* addr, std::size_t bytes,
216  std::size_t alignment)
217  {
218  if (addr == nullptr)
219  {
220  trace::printf ("%s(0) nop\n", __func__);
221  return;
222  }
223 
224 #if defined(OS_TRACE_LIBC_MALLOC__)
225  trace::printf ("%s(%p,%u,%u)\n", __func__, p, bytes,alignment);
226 #endif
227 
228  // Compute the chunk address from the user address.
229  chunk_t* chunk = reinterpret_cast<chunk_t *> (static_cast<char *> (addr)
230  - chunk_offset);
231 #if 0
232  if (chunk->size < 0)
233  {
234  chunk = static_cast<chunk_t *> (static_cast<char *> (chunk)
235  + chunk->size);
236  }
237 #endif
238 
239  // If the free list is empty, create it with the current chunk, alone.
240  if (free_list_ == nullptr)
241  {
242  // Mark the end of the list with a null pointer.
243  chunk->next = nullptr;
244 
245  // The chunk becomes the first list element.
246  free_list_ = chunk;
247  return;
248  }
249 
250  // The free list exists; is the chunk before the list head?
251  if (chunk < free_list_)
252  {
253  // Is the chunk *right* before the list head?
254  if (reinterpret_cast<char *> (chunk) + chunk->size
255  == reinterpret_cast<char *> (free_list_))
256  {
257  // Coalesce chunk to the list head.
258  chunk->size += free_list_->size;
259  chunk->next = free_list_->next;
260  }
261  else
262  {
263  // Insert before the list head.
264  chunk->next = free_list_;
265  }
266  // The chunk becomes the new list head.
267  free_list_ = chunk;
268  return;
269  }
270 
271  // Walk the free list to find the place to insert,
272  // (the list must be ordered by addresses).
273  // Warning: not deterministic!
274 
275  chunk_t* q = free_list_;
276  chunk_t* p;
277  do
278  {
279  p = q;
280  q = q->next;
281  }
282  while (q != nullptr && q <= chunk);
283 
284  // Now p <= chunk and either q == nullptr or q > chunk.
285  // Try to merge with chunks immediately before/after it.
286 
287  if (reinterpret_cast<char *> (p) + p->size
288  == reinterpret_cast<char *> (chunk))
289  {
290  // Chunk to be freed is adjacent to a free chunk before it.
291  p->size += chunk->size;
292  // If the merged chunk is also adjacent to the chunk after it,
293  // merge again.
294  if (reinterpret_cast<char *> (p) + p->size
295  == reinterpret_cast<char *> (q))
296  {
297  p->size += q->size;
298  p->next = q->next;
299  }
300  }
301  else if (reinterpret_cast<char *> (p) + p->size
302  > reinterpret_cast<char *> (chunk))
303  {
304  // Already freed.
305  return;
306  }
307  else if (reinterpret_cast<char *> (chunk) + chunk->size
308  == reinterpret_cast<char *> (q))
309  {
310  // Chunk to be freed is adjacent to a free chunk after it.
311  chunk->size += q->size;
312  chunk->next = q->next;
313  p->next = chunk;
314  }
315  else
316  {
317  // Not adjacent to any chunk. Just insert it.
318  // The result is a new fragment. Not great...
319  chunk->next = q;
320  p->next = chunk;
321  }
322  }
