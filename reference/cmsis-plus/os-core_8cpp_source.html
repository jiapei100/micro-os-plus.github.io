377 #if defined(OS_INCLUDE_RTOS_STATISTICS_THREAD_CPU_CYCLES)
378 
379  // Get the high resolution timestamp.
380  clock::timestamp_t now = hrclock.now ();
381 
382  // Compute duration since previous context switch.
383  // Assume scheduler is not disabled for very long.
384  rtos::statistics::duration_t delta =
385  static_cast<rtos::statistics::duration_t> (now
386  - scheduler::statistics::switch_timestamp_);
387 
388  // Accumulate durations to scheduler total.
389  scheduler::statistics::cpu_cycles_ += delta;
390 
391  // Accumulate durations to old thread.
392  scheduler::current_thread_->statistics_.cpu_cycles_ += delta;
393 
394  // Remember the timestamp for the next context switch.
395  scheduler::statistics::switch_timestamp_ = now;
396 
397 #endif /* defined(OS_INCLUDE_RTOS_STATISTICS_THREAD_CPU_CYCLES) */
398 
399  // Normally the old running thread must be re-linked to ready.
400  scheduler::current_thread_->internal_relink_running_ ();
401 
402  // The top of the ready list gives the next thread to run.
403  scheduler::current_thread_ =
404  scheduler::ready_threads_list_.unlink_head ();
405 
406  // ***** Pointer switched to new thread! *****
407 
408  // The new thread was marked as running in unlink_head(),
409  // so in case the handler is re-entered immediately,
410  // the relink_running() will simply reschedule it,
411  // otherwise the thread will be lost.
