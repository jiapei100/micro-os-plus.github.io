394  void
395  _switch_threads (void)
396  {
397 #if defined(OS_INCLUDE_RTOS_STATISTICS_THREAD_CPU_CYCLES)
398 
399  // Get the high resolution timestamp.
400  clock::timestamp_t now = hrclock.now ();
401 
402  // Compute duration since previous context switch.
403  // Assume scheduler is not disabled for very long.
404  rtos::statistics::duration_t delta =
405  static_cast<rtos::statistics::duration_t> (now
406  - scheduler::statistics::switch_timestamp_);
407 
408  // Accumulate durations to scheduler total.
409  scheduler::statistics::cpu_cycles_ += delta;
410 
411  // Accumulate durations to old thread.
412  scheduler::current_thread_->statistics_.cpu_cycles_ += delta;
413 
414  // Remember the timestamp for the next context switch.
415  scheduler::statistics::switch_timestamp_ = now;
416 
417 #endif /* defined(OS_INCLUDE_RTOS_STATISTICS_THREAD_CPU_CYCLES) */
418 
419  // Normally the old running thread must be re-linked to ready.
420  scheduler::current_thread_->_relink_running ();
421 
422  // The top of the ready list gives the next thread to run.
423  scheduler::current_thread_ =
424  scheduler::ready_threads_list_.unlink_head ();
425 
426  // ***** Pointer switched to new thread! *****
427 
428  // The new thread was marked as running in unlink_head(),
429  // so in case the handler is re-entered immediately,
430  // the relink_running() will simply reschedule it,
431  // otherwise the thread will be lost.
432 
433 #if defined(OS_INCLUDE_RTOS_STATISTICS_THREAD_CONTEXT_SWITCHES)
434 
435  // Increment global context switches.
436  scheduler::statistics::context_switches_++;
437 
438  // Increment new thread context switches.
439  scheduler::current_thread_->statistics_.context_switches_++;
440 
441 #endif /* defined(OS_INCLUDE_RTOS_STATISTICS_THREAD_CONTEXT_SWITCHES) */
442 
443  }
