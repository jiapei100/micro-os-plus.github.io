192  void
193  Ready_threads_list::link (Waiting_thread_node& node)
194  {
195  if (head_.prev == nullptr)
196  {
197  // If this is the first time, initialise the list to empty.
198  clear ();
199  }
200 
201  thread::priority_t prio = node.thread.prio_;
202 
203  Waiting_thread_node* after =
204  static_cast<Waiting_thread_node*> (const_cast<Static_double_list_links *> (tail ()));
205 
206  if (empty () || (prio <= after->thread.prio_))
207  {
208  // Insert at the end of the list.
209  }
210  else if (prio > head ()->thread.prio_)
211  {
212  // Insert at the beginning of the list
213  after =
214  static_cast<Waiting_thread_node*> (const_cast<Static_double_list_links *> (&head_));
215 #if defined(OS_TRACE_RTOS_LISTS)
216  trace::printf ("ready %s() head \n", __func__);
217 #endif
218  }
219  else
220  {
221  // Insert in the middle of the list.
222  // The loop is guaranteed to terminate, not hit the head and
223  // the weight is small, sched_prio() is only an accessor.
224  while (prio > after->thread.prio_)
225  {
226  after =
227  static_cast<Waiting_thread_node*> (const_cast<Static_double_list_links *> (after->prev));
228  }
229  }
230 
231 #if defined(OS_TRACE_RTOS_LISTS)
232  trace::printf ("ready %s() %p %s\n", __func__, &node.thread,
233  node.thread.name ());
234 #endif
235 
236  insert_after (node, after);
237 
238  node.thread.sched_state_ = thread::state::ready;
239  }

355  void
356  Waiting_threads_list::resume_one (void)
357  {
358  Thread* thread;
359  {
360  interrupts::Critical_section ics; // ----- Critical section -----
361 
362  // If the list is empty, silently return.
363  if (empty ())
364  {
365  return;
366  }
367 
368  // The top priority is to remove the entry from the list
369  // so that subsequent wakeups to address different threads.
370  thread = &(head ()->thread);
371  const_cast<Waiting_thread_node*> (head ())->unlink ();
372  }
373  assert(thread != nullptr);
374 
375  thread::state_t state = thread->sched_state ();
376  if (state != thread::state::destroyed)
377  {
378  thread->resume ();
379  }
380  else
381  {
382 #if defined(OS_TRACE_RTOS_LISTS)
383  trace::printf ("%s() gone \n", __func__);
384 #endif
385  }

555  void
556  Clock_timestamps_list::check_timestamp (clock::timestamp_t now)
557  {
558  if (head_.next == nullptr)
559  {
560  // This happens before the constructors are executed.
561  return;
562  }
563 
564  // Multiple threads can wait for the same time stamp, so
565  // iterate until a node with future time stamp is identified.
566  for (;;)
567  {
568  interrupts::Critical_section ics; // ----- Critical section -----
569 
570  if (empty ())
571  {
572  break;
573  }

575  if (now >= head_ts)
