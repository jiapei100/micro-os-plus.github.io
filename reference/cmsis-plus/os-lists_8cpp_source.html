100  Waiting_threads_list::add (DoubleListNodeThread& node)
101  {
102  if (head_ == nullptr)
103  {
104  // Make the node point to itself, to satisfy the double
105  // linked circular list.
106  node.prev = &node;
107  node.next = &node;
108 
109  head_ = &node;
110  count_ = 1;
111  }
112  else
113  {
114  thread::priority_t prio = node.node.sched_prio ();
115 
116  DoubleListLinks* after = (head_->prev);
117 
118  if (prio
119  <= ((DoubleListNodeThread*) (head_->next))->node.sched_prio ())
120  {
121  // Insert at the end of the list.
122  }
123  else if (prio > ((DoubleListNodeThread*) (head_))->node.sched_prio ())
124  {
125  // Insert at the beginning of the list
126  // and update the new head.
127  head_ = &node;
128  }
129  else
130  {
131  // Insert in the middle of the list.
132  // The loop is guaranteed to terminate and the
133  // weight is small, sched_prio() is only an accessor.
134  while (prio
135  > ((DoubleListNodeThread*) (after))->node.sched_prio ())
136  {
137  after = after->prev;
138  }
139  }
140 
141  // Make the new node point to its neighbours.
142  node.prev = after;
143  node.next = after->next;
144 
145  // Make the neighbours point to the node. The order is important.
146  after->next->prev = &node;
147  after->next = &node;
148 
149  ++count_;
150  }
151  }

163  Waiting_threads_list::remove (DoubleListNodeThread& node)
164  {
165  // Check if not already removed.
166  if (node.next == nullptr)
167  {
168  return;
169  }
170 
171  if (count_ > 1)
172  {
173  if (head_ == &node)
174  {
175  // Move head to next node.
176  head_ = (DoubleListNodeThread*) node.next;
177  }
178 
179  // Make neighbours point to each other.
180  node.prev->next = node.next;
181  node.next->prev = node.prev;
182 
183  --count_;
184  }
185  else if (count_ == 1)
186  {
187  clear ();
188  }
189 
190  // Nullify both pointers in the removed node.
191  node.prev = nullptr;
192  node.next = nullptr;
193  }
