81  void
82  Double_list::remove (Double_list_links& node)
83  {
84  // Check if not already removed.
85  if (node.next == nullptr)
86  {
87  return;
88  }
89 
90  if (count_ > 1)
91  {
92  if (head_ == &node)
93  {
94  // Move head to next node, this one will vanish.
95  head_ = (Double_list_links*) node.next;
96  }
97 
98  // Make neighbours point to each other.
99  node.prev->next = node.next;
100  node.next->prev = node.prev;
101 
102  --count_;
103  }
104  else if (count_ == 1)
105  {
106  clear ();
107  }
108 
109  // Nullify both pointers in the removed node.
110  node.prev = nullptr;
111  node.next = nullptr;
112  }

157  void
158  Waiting_threads_list::add (Waiting_thread_node& node)
159  {
160  if (head_ == nullptr)
161  {
162  // Make the node point to itself, to satisfy the double
163  // linked circular list.
164  node.prev = &node;
165  node.next = &node;
166 
167  head_ = &node;
168  count_ = 1;
169  }
170  else
171  {
172  thread::priority_t prio = node.thread.sched_prio ();
173 
174  Waiting_thread_node* after = (Waiting_thread_node*) (head_->prev);
175 
176  if (prio <= after->thread.sched_prio ())
177  {
178  // Insert at the end of the list.
179  }
180  else if (prio > head ()->thread.sched_prio ())
181  {
182  // Insert at the beginning of the list
183  // and update the new head.
184  head_ = &node;
185  trace::printf ("%s() head \n", __func__);
186  }
187  else
188  {
189  // Insert in the middle of the list.
190  // The loop is guaranteed to terminate and the
191  // weight is small, sched_prio() is only an accessor.
192  while (prio > after->thread.sched_prio ())
193  {
194  after = (Waiting_thread_node*) after->prev;
195  }
196  }
197 
198  // Make the new node point to its neighbours.
199  node.prev = after;
200  node.next = after->next;
201 
202  // Make the neighbours point to the n. The order is important.
203  after->next->prev = &node;
204  after->next = &node;
205 
206  ++count_;
207  }
208  }

215  void
216  Waiting_threads_list::resume_one (void)
217  {
218  Thread* thread;
219  {
220  interrupts::Critical_section cs; // ----- Critical section -----
221 
222  // If the list is empty, silently return.
223  if (empty ())
224  {
225  return;
226  }
227 
228  // The top priority is to remove the entry from the list
229  // so that subsequent wakeups to address different threads.
230  thread = &(head ()->thread);
231  remove (*head ());
232  }
233  assert(thread != nullptr);
234 
235  thread::state_t state = thread->sched_state ();
236  if (state != thread::state::destroyed)
237  {
238  thread->resume ();
239  }
240  else
241  {
242  trace::printf ("%s() gone \n", __func__);
243  }

346  Clock_timestamps_list::add (Timestamp_node& node)
347  {
348  if (head_ == nullptr)
349  {
350  // Make the node point to itself, to satisfy the double
351  // linked circular list.
352  node.prev = &node;
353  node.next = &node;
354 
355  trace::printf ("%s() %u \n", __func__, (uint32_t) node.timestamp);
356 
357  head_ = &node;
358  count_ = 1;
359  }
360  else
361  {
362  clock::timestamp_t timestamp = node.timestamp;
363 
364  Timeout_thread_node* after = (Timeout_thread_node*) (head_->prev);
365 
366  if (timestamp >= after->timestamp)
367  {
368  // Insert at the end of the list.
369  }
370  else if (timestamp < head ()->timestamp)
371  {
372  // Insert at the beginning of the list
373  // and update the new head.
374  head_ = &node;
375  trace::printf ("%s() head \n", __func__);
376  }
377  else
378  {
379  // Insert in the middle of the list.
380  // The loop is guaranteed to terminate.
381  while (timestamp < after->timestamp)
382  {
383  after = (Timeout_thread_node*) (after->prev);
384  }
385  }
386 
387  ++count_;
388 
389  trace::printf ("%s() %u after %u #%d\n", __func__,
390  (uint32_t) timestamp, (uint32_t) after->timestamp,
391  count_);
392 
393  assert(after->timestamp != 0);
394 
395  // Make the new node point to its neighbours.
396  node.prev = after;
397  node.next = after->next;
398 
399  // Make the neighbours point to the node. The order is important.
400  after->next->prev = &node;
401  after->next = &node;
402  }
