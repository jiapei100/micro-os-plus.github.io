142  Waiting_threads_list::link (Waiting_thread_node& node)
143  {
144  thread::priority_t prio = node.thread.sched_prio ();
145 
146  Waiting_thread_node* after = (Waiting_thread_node*) (head_.prev);
147 
148  if (empty () || (prio <= after->thread.sched_prio ()))
149  {
150  // Insert at the end of the list.
151  }
152  else if (prio > head ()->thread.sched_prio ())
153  {
154  // Insert at the beginning of the list
155  after = (Waiting_thread_node*) (&head_);
156 #if defined(OS_TRACE_RTOS_LISTS)
157  trace::printf ("%s() head \n", __func__);
158 #endif
159  }
160  else
161  {
162  // Insert in the middle of the list.
163  // The loop is guaranteed to terminate, not hit the head and
164  // the weight is small, sched_prio() is only an accessor.
165  while (prio > after->thread.sched_prio ())
166  {
167  after = (Waiting_thread_node*) after->prev;
168  }
169  }
170 
171  // Make the new node point to its neighbours.
172  node.prev = after;
173  node.next = after->next;
174 
175  // Make the neighbours point to the n. The order is important.
176  after->next->prev = &node;
177  after->next = &node;
178  }

185  void
186  Waiting_threads_list::resume_one (void)
187  {
188  Thread* thread;
189  {
190  interrupts::Critical_section ics; // ----- Critical section -----
191 
192  // If the list is empty, silently return.
193  if (empty ())
194  {
195  return;
196  }
197 
198  // The top priority is to remove the entry from the list
199  // so that subsequent wakeups to address different threads.
200  thread = &(head ()->thread);
201  head ()->unlink ();
202  }
203  assert(thread != nullptr);
204 
205  thread::state_t state = thread->sched_state ();
206  if (state != thread::state::destroyed)
207  {
208  thread->resume ();
209  }
210  else
211  {
212 #if defined(OS_TRACE_RTOS_LISTS)
213  trace::printf ("%s() gone \n", __func__);
214 #endif
215  }
216  }

314  void
315  Clock_timestamps_list::link (Timestamp_node& node)
316  {
317  clock::timestamp_t timestamp = node.timestamp;
318 
319  Timeout_thread_node* after = (Timeout_thread_node*) (head_.prev);
320 
321  if (empty () || (timestamp >= after->timestamp))
322  {
323  // Insert at the end of the list.
324  }
325  else if (timestamp < head ()->timestamp)
326  {
327  // Insert at the beginning of the list
328  // and update the new head.
329  after = (Timeout_thread_node*) (&head_);
330 #if defined(OS_TRACE_RTOS_LISTS)
331  trace::printf ("%s() head \n", __func__);
332 #endif
333  }
334  else
335  {
336  // Insert in the middle of the list.
337  // The loop is guaranteed to terminate.
338  while (timestamp < after->timestamp)
339  {
340  after = (Timeout_thread_node*) (after->prev);
341  }
342  }
343 
344 #if defined(OS_TRACE_RTOS_LISTS)
345  trace::printf ("%s() %u\n", __func__, (uint32_t) timestamp);
346 #endif
347 
348  // Make the new node point to its neighbours.
349  node.prev = after;
350  node.next = after->next;
351 
352  // Make the neighbours point to the node. The order is important.
353  after->next->prev = &node;
354  after->next = &node;
355  }

364  void
365  Clock_timestamps_list::check_timestamp (clock::timestamp_t now)
366  {
367  if (head_.next == nullptr)
368  {
369  // This happens before the constructors are executed.
370  return;
371  }
372 
373  // Multiple threads can wait for the same time stamp, so
374  // iterate until a node with future time stamp is identified.
375  for (; !empty ();)
376  {
377  clock::timestamp_t head_ts = head ()->timestamp;
378  if (now >= head_ts)
379  {
380 #if defined(OS_TRACE_RTOS_LISTS)
381  trace::printf ("%s() %u \n", __func__,
382  (uint32_t) systick_clock.now ());
383 #endif
384  head ()->action ();
385  }
386  else
387  {
388  break;
389  }
390  }
391  }
