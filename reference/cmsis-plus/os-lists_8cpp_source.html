192  void
193  ready_threads_list::link (waiting_thread_node& node)
194  {
195  if (head_.prev () == nullptr)
196  {
197  // If this is the first time, initialise the list to empty.
198  clear ();
199  }
200 
201  thread::priority_t prio = node.thread_.prio_;
202 
203  waiting_thread_node* after =
204  static_cast<waiting_thread_node*> (const_cast<static_double_list_links *> (tail ()));
205 
206  if (empty ())
207  {
208  // Insert at the end of the list.
209 #if defined(OS_TRACE_RTOS_LISTS)
210  trace::printf ("ready %s() empty +%u\n", __func__, prio);
211 #endif
212  }
213  else if (prio <= after->thread_.prio_)
214  {
215  // Insert at the end of the list.
216 #if defined(OS_TRACE_RTOS_LISTS)
217  trace::printf ("ready %s() back %u +%u \n", __func__,
218  after->thread_.prio_, prio);
219 #endif
220  }
221  else if (prio > head ()->thread_.prio_)
222  {
223  // Insert at the beginning of the list
224  after =
225  static_cast<waiting_thread_node*> (const_cast<static_double_list_links *> (&head_));
226 #if defined(OS_TRACE_RTOS_LISTS)
227  trace::printf ("ready %s() front +%u %u \n", __func__, prio,
228  head ()->thread_.prio_);
229 #endif
230  }
231  else
232  {
233  // Insert in the middle of the list.
234  // The loop is guaranteed to terminate, not hit the head and
235  // the weight is small, priority() is only an accessor.
236  while (prio > after->thread_.prio_)
237  {
238  after =
239  static_cast<waiting_thread_node*> (const_cast<static_double_list_links *> (after->prev ()));
240  }
241 #if defined(OS_TRACE_RTOS_LISTS)
242  trace::printf ("ready %s() middle %u +%u \n", __func__,
243  after->thread_.prio_, prio);
244 #endif
245  }
246 
247  insert_after (node, after);
248 
249  node.thread_.state_ = thread::state::ready;
250  }

380  void
381  waiting_threads_list::resume_one (void)
382  {
383  thread* thread;
384  {
385  interrupts::critical_section ics; // ----- Critical section -----
386 
387  // If the list is empty, silently return.
388  if (empty ())
389  {
390  return;
391  }
392 
393  // The top priority is to remove the entry from the list
394  // so that subsequent wakeups to address different threads.
395  thread = &(head ()->thread_);
396  const_cast<waiting_thread_node*> (head ())->unlink ();
397  }
398  assert(thread != nullptr);
399 
400  thread::state_t state = thread->state ();
401  if (state != thread::state::destroyed)
402  {
403  thread->resume ();
404  }
405  else
406  {
407 #if defined(OS_TRACE_RTOS_LISTS)
408  trace::printf ("%s() gone \n", __func__);
409 #endif
410  }

523  void
524  clock_timestamps_list::link (timestamp_node& node)
525  {
526  clock::timestamp_t timestamp = node.timestamp;
527 
528  timeout_thread_node* after =
529  static_cast<timeout_thread_node*> (const_cast<static_double_list_links *> (tail ()));
530 
531  if (empty ())
532  {
533  // Insert at the end of the list.
534 #if defined(OS_TRACE_RTOS_CLOCKS)
535  trace::printf ("clock %s() empty +%u\n", __func__,
536  static_cast<uint32_t> (timestamp));
537 #endif
538  }
539  else if (timestamp >= after->timestamp)
540  {
541  // Insert at the end of the list.
542 #if defined(OS_TRACE_RTOS_CLOCKS)
543  trace::printf ("clock %s() back %u +%u\n", __func__,
544  static_cast<uint32_t> (after->timestamp),
545  static_cast<uint32_t> (timestamp));
546 #endif
547  }
548  else if (timestamp < head ()->timestamp)
549  {
550  // Insert at the beginning of the list
551  // and update the new head.
552  after =
553  static_cast<timeout_thread_node*> (const_cast<static_double_list_links *> (&head_));
554 #if defined(OS_TRACE_RTOS_CLOCKS)
555  trace::printf ("clock %s() front +%u %u\n", __func__,
556  static_cast<uint32_t> (timestamp),
557  static_cast<uint32_t> (head ()->timestamp));
558 #endif
559  }
560  else
561  {
562  // Insert in the middle of the list.
563  // The loop is guaranteed to terminate.
564  while (timestamp < after->timestamp)
565  {
566  after =
567  static_cast<timeout_thread_node*> (const_cast<static_double_list_links *> (after->prev ()));
568  }
569 #if defined(OS_TRACE_RTOS_CLOCKS)
570  trace::printf ("clock %s() middle %u +%u\n", __func__,
571  static_cast<uint32_t> (after->timestamp),
572  static_cast<uint32_t> (timestamp));
573 #endif
574  }
575 
576  insert_after (node, after);
577  }

586  void
587  clock_timestamps_list::check_timestamp (clock::timestamp_t now)
588  {
589  if (head_.next () == nullptr)
590  {
591  // This happens before the constructors are executed.
592  return;
593  }
594 
595  // Multiple threads can wait for the same time stamp, so
596  // iterate until a node with future time stamp is identified.
597  for (;;)
598  {
599  interrupts::critical_section ics; // ----- Critical section -----
600 
601  if (empty ())
602  {
603  break;
604  }
605  clock::timestamp_t head_ts = head ()->timestamp;
606  if (now >= head_ts)
607  {
608 #if defined(OS_TRACE_RTOS_CLOCKS)
609  trace::printf ("%s() %u \n", __func__,
610  static_cast<uint32_t> (sysclock.now ()));
611 #endif
612  const_cast<timestamp_node*> (head ())->action ();
613  }
614  else
615  {
616  break;
617  }
618  }
619  }
