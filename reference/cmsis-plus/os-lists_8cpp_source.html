73  Double_list::remove (Double_list_links& node)
74  {
75  // Check if not already removed.
76  if (node.next == nullptr)
77  {
78  return;
79  }
80 
81  if (count_ > 1)
82  {
83  if (head_ == &node)
84  {
85  // Move head to next node.
86  head_ = (Double_list_links*) node.next;
87  }
88 
89  // Make neighbours point to each other.
90  node.prev->next = node.next;
91  node.next->prev = node.prev;
92 
93  --count_;
94  }
95  else if (count_ == 1)
96  {
97  clear ();
98  }
99 
100  // Nullify both pointers in the removed node.
101  node.prev = nullptr;
102  node.next = nullptr;
103  }

148  void
149  Waiting_threads_list::add (Double_list_node_thread& node)
150  {
151  if (head_ == nullptr)
152  {
153  // Make the n point to itself, to satisfy the double
154  // linked circular list.
155  node.prev = &node;
156  node.next = &node;
157 
158  head_ = &node;
159  count_ = 1;
160  }
161  else
162  {
163  thread::priority_t prio = node.thread.sched_prio ();
164 
165  Double_list_node_thread* after =
166  (Double_list_node_thread*) (head_->prev);
167 
168  if (prio <= after->thread.sched_prio ())
169  {
170  // Insert at the end of the list.
171  }
172  else if (prio > head ()->thread.sched_prio ())
173  {
174  // Insert at the beginning of the list
175  // and update the new head.
176  head_ = &node;
177  trace::printf ("%s() head \n", __func__);
178  }
179  else
180  {
181  // Insert in the middle of the list.
182  // The loop is guaranteed to terminate and the
183  // weight is small, sched_prio() is only an accessor.
184  while (prio > after->thread.sched_prio ())
185  {
186  after = (Double_list_node_thread*) after->prev;
187  }
188  }
189 
190  // Make the new node point to its neighbours.
191  node.prev = after;
192  node.next = after->next;
193 
194  // Make the neighbours point to the n. The order is important.
195  after->next->prev = &node;
196  after->next = &node;
197 
198  ++count_;
199  }
200  }

346  Clock_timestamps_list::add (Double_list_node_timestamp& node)
347  {
348  if (head_ == nullptr)
349  {
350  // Make the node point to itself, to satisfy the double
351  // linked circular list.
352  node.prev = &node;
353  node.next = &node;
354 
355  trace::printf ("%s() %u \n", __func__, (uint32_t) node.timestamp);
356 
357  head_ = &node;
358  count_ = 1;
359  }
360  else
361  {
362  clock::timestamp_t timestamp = node.timestamp;
363 
364  Double_list_node_clock* after =
365  (Double_list_node_clock*) (head_->prev);
366 
367  if (timestamp >= after->timestamp)
368  {
369  // Insert at the end of the list.
370  }
371  else if (timestamp < head ()->timestamp)
372  {
373  // Insert at the beginning of the list
374  // and update the new head.
375  head_ = &node;
376  trace::printf ("%s() head \n", __func__);
377  }
378  else
379  {
380  // Insert in the middle of the list.
381  // The loop is guaranteed to terminate.
382  while (timestamp < after->timestamp)
383  {
384  after = (Double_list_node_clock*) (after->prev);
385  }
386  }
387 
388  ++count_;
389

391  (uint32_t) timestamp, (uint32_t) after->timestamp,

393
