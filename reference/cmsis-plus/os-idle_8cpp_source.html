58  // The thread was created with the default priority, and the
59  // idle thread must run with th lowest possible priority.

61 #if defined(OS_BOOL_RTOS_THREAD_IDLE_PRIORITY_BELOW_IDLE)
62  // The CMSIS RTOS validator creates threads with `priority::idle`,
63  // so, to be sure that the system idle thread has the lowest priority,
64  // go one step below the idle priority.
65  this_thread::thread ().priority (thread::priority::idle - 1);
66 #else

68 #endif

70  while (true)
