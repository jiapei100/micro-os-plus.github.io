569  * Internal function.

572  result_t
573  mutex::internal_try_lock_ (thread* crt_thread)
574  {
575  // Save the initial owner for later protocol tests.
576  thread* saved_owner = owner_;
577 
578  // First lock.
579  if (owner_ == nullptr)
580  {
581  // If the mutex has no owner, own it.
582  owner_ = crt_thread;
583 
584  // For recursive mutexes, initialise counter.
585  count_ = 1;
586 
587  // When the mutex is acquired, some more actions are
588  // required, according to mutex attributes.
589 
590  if (robustness_ == robustness::robust)
591  {
592  mutexes_list* th_list =
593  reinterpret_cast<mutexes_list*> (&owner_->mutexes_);
594  th_list->link (*this);
595  }
596  else
597  {
598  // Count the number of non-robust mutexes acquired by the thread.
599  // Terminating a thread with locked non-robust mutexes
600  // will trigger an assert.
601  ++(crt_thread->acquired_mutexes_);
602  }
603 
604  if (protocol_ == protocol::protect)
605  {
606  if (crt_thread->priority () > prio_ceiling_)
607  {
608  // Prio ceiling must be at least the priority of the
609  // highest priority thread.
610  return EINVAL;
611  }
612 
613  // POSIX: When a thread owns one or more mutexes
614  // initialised with the mutex::protocol::protect protocol,
615  // it shall execute at the higher of its priority or the
616  // highest of the priority ceilings of all the mutexes
617  // owned by this thread and initialised with this
618  // attribute, regardless of whether other threads are
619  // blocked on any of these robust mutexes or not.
620 
621  // Boost priority.
622  boosted_prio_ = prio_ceiling_;
623  if (boosted_prio_ > owner_->priority_inherited ())
624  {
625  // ----- Enter uncritical section ---------------------------
626  scheduler::uncritical_section sucs;
627 
628  owner_->priority_inherited (boosted_prio_);
629  // ----- Exit uncritical section ----------------------------
630  }
631  }
632 
633 #if defined(OS_TRACE_RTOS_MUTEX)
634  trace::printf ("%s() @%p %s by %p %s LCK\n", __func__, this, name (),
635  crt_thread, crt_thread->name ());
636 #endif
637  // If the owning thread of a robust mutex terminates while
638  // holding the mutex lock, the next thread that acquires the
639  // mutex may be notified about the termination by the return
640  // value EOWNERDEAD.
641  if (owner_dead_)
642  {
643  return EOWNERDEAD;
644  }
645  return result::ok;
646  }
647 
648  // Relock? (lock by the same thread).
649  if (saved_owner == crt_thread)
650  {
651  // The mutex was requested again by the same thread.
652  if (type_ == type::recursive)
653  {
654  if (count_ >= max_count_)
655  {
656  // The recursive mutex reached its limit.
657 #if defined(OS_TRACE_RTOS_MUTEX)
658  trace::printf ("%s() @%p %s EAGAIN\n", __func__, this,
659  name ());
660 #endif
661  return EAGAIN;
662  }
663 
664  // Increment the recursion depth counter.
665  ++count_;
666 
667 #if defined(OS_TRACE_RTOS_MUTEX)
668  trace::printf ("%s() @%p %s by %p %s >%u\n", __func__, this,
669  name (), crt_thread, crt_thread->name (), count_);
670 #endif
671  return result::ok;
672  }
673  else if (type_ == type::errorcheck)
674  {
675  // Errorcheck mutexes do not block, but return an error.
676 #if defined(OS_TRACE_RTOS_MUTEX)
677  trace::printf ("%s() @%p %s EDEADLK\n", __func__, this, name ());
678 #endif
679  return EDEADLK;
680  }
681  else if (type_ == type::normal)
682  {
683 #if defined(OS_TRACE_RTOS_MUTEX)
684  trace::printf ("%s() @%p %s deadlock\n", __func__, this, name ());
685 #endif
686  return EWOULDBLOCK;
687  }
688 
689  return EWOULDBLOCK;
690  }
691  else
692  {
693  // Try to lock when not owner (another thread requested the mutex).
694 
695  // POSIX: When a thread makes a call to mutex::lock(), the mutex was
696  // initialised with the protocol attribute having the value
697  // mutex::protocol::inherit, when the calling thread is blocked
698  // because the mutex is owned by another thread, that owner thread
699  // shall inherit the priority level of the calling thread as long
700  // as it continues to own the mutex. The implementation shall
701  // update its execution priority to the maximum of its assigned
702  // priority and all its inherited priorities.
703  // Furthermore, if this owner thread itself becomes blocked on
704  // another mutex with the protocol attribute having the value
705  // mutex::protocol::inherit, the same priority inheritance effect
706  // shall be propagated to this other owner thread, in a recursive
707  // manner.
708  if (protocol_ == protocol::inherit)
709  {
710  thread::priority_t prio = crt_thread->priority ();
711  boosted_prio_ = prio;
712 
713  mutexes_list* th_list =
714  reinterpret_cast<mutexes_list*> (&owner_->mutexes_);
715  if (owner_links_.unlinked ())
716  {
717  th_list->link (*this);
718  }
719 
720  // Boost owner priority.
721  if ((boosted_prio_ > owner_->priority_inherited ()))
722  {
723  // ----- Enter uncritical section ---------------------------
724  scheduler::uncritical_section sucs;
725 
726  owner_->priority_inherited (boosted_prio_);
727  // ----- Exit uncritical section ----------------------------
728  }
729 
730 #if defined(OS_TRACE_RTOS_MUTEX)
731  trace::printf ("%s() @%p %s boost %u by %p %s \n", __func__, this,
732  name (), boosted_prio_, crt_thread,
733  crt_thread->name ());
734 #endif
735 
736  return EWOULDBLOCK;
737  }
738  }
739 
740  // Block anyway.
741  return EWOULDBLOCK;
742  }

1189  scheduler::critical_section scs;

1192  if (owner_ == crt_thread)
1193  {

1195  {
1196  --count_;

1202  }

1205  {

1208

1210  // not linked.
1211  owner_links_.unlink ();
1212

1214  {

1216  reinterpret_cast<mutexes_list*> (&owner_->mutexes_);
1217

1222  // and the assigned priority will take precedence.
1223  boosted_prio_ = thread::priority::none;

1238  }

1240  owner_->priority_inherited (boosted_prio_);
