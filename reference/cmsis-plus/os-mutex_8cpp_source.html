488  result_t
489  Mutex::_try_lock (Thread* crt_thread)
490  {
491  Thread* saved_owner;
492 
493  {
494  scheduler::Critical_section cs; // ----- Critical section -----
495 
496  saved_owner = owner_;
497  if (owner_ == nullptr)
498  {
499  owner_ = crt_thread;
500  count_ = 1;
501  }
502  // Short critical section ends here.
503  }
504 
505  if (saved_owner == nullptr)
506  {
507  if (protocol_ == mutex::protocol::inherit)
508  {
509  // Save owner priority, in case a temporary boost
510  // will be later applied.
511  owner_prio_ = owner_->sched_prio ();
512  }
513  else if (protocol_ == mutex::protocol::protect)
514  {
515  // Save owner priority and boost priority.
516  owner_prio_ = owner_->sched_prio ();
517  if (prio_ceiling_ > owner_prio_)
518  {
519  owner_->sched_prio (prio_ceiling_);
520  }
521  }
522  trace::printf ("mutex @%p %s locked by %p %s\n", this, name (),
523  crt_thread, crt_thread->name ());
524  return result::ok;
525  }
526 
527  if (saved_owner == crt_thread)
528  {
529  if (type_ == mutex::type::recursive)
530  {
531  if (count_ == max_count_)
532  {
533  return EAGAIN;
534  }
535  ++count_;
536  trace::printf ("mutex @%p %s incr %d by %p %s\n", this, name (),
537  count_, crt_thread, crt_thread->name ());
538  return result::ok;
539  }
540  else if (type_ == mutex::type::errorcheck)
541  {
542  return EDEADLK;
543  }
544  }
545 
546  if (protocol_ == mutex::protocol::inherit)
547  {
548  thread::priority_t prio = crt_thread->sched_prio ();
549  if ((prio > owner_->sched_prio ()))
550  {
551  // Boost owner priority.
552  owner_->sched_prio (prio);
553  }
554  }
555 
556  return EBUSY;
557 
558  // TODO: EINVAL, EOWNERDEAD
559  }

603  Mutex::lock (void)
604  {
605  os_assert_err(!scheduler::in_handler_mode (), EPERM);
606 
607  trace::printf ("%s() @%p %s by %p %s\n", __func__, this, name (),
608  &this_thread::thread (), this_thread::thread ().name ());
609 
610 #if defined(OS_INCLUDE_RTOS_PORT_MUTEX)
611 
612  return port::Mutex::lock (this);
613 
614 #else
615 
616  Thread& crt_thread = this_thread::thread ();
617 
618  // Prepare a list node pointing to the current thread.
619  // Do not worry for being on stack, it is temporarily linked to the
620  // list and guaranteed to be removed before this function returns.
621  Waiting_thread_node node
622  { list_, crt_thread };
623 
624  for (;;)
625  {
626 
627  result_t res = _try_lock (&crt_thread);
628  if (res != EBUSY)
629  {
630  return res;

632

635  // It is removed when this block ends (after sleep()).
636  Waiting_threads_list_guard<scheduler::Critical_section> lg

638

640  }
641

643  {
644  return EINTR;
645  }
646  }
