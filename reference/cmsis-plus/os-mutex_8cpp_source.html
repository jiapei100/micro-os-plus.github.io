568  mutex::_try_lock (thread* crt_thread)
569  {
570  // TODO: implement robust mutex and return ENOTRECOVERABLE, EOWNERDEAD.
571 
572  // Save the initial owner for later protocol tests.
573  thread* saved_owner = owner_;
574 
575  // First lock.
576  if (owner_ == nullptr)
577  {
578  // If the mutex has no owner, own it.
579  owner_ = crt_thread;
580 
581  // For recursive mutexes, initialise counter.
582  count_ = 1;
583 
584  // Count the number of mutexes acquired by the thread.
585  ++(crt_thread->acquired_mutexes_);
586 
587  // When the mutex is acquired, some more actions are
588  // required, according to mutex type.
589  if (protocol_ == protocol::inherit)
590  {
591  // Save owner priority, in case a temporary boost
592  // will be later applied.
593  owner_prio_ = owner_->priority ();
594  }
595  else if (protocol_ == protocol::protect)
596  {
597  if (crt_thread->priority () > prio_ceiling_)
598  {
599  return EINVAL;
600  }
601 
602  // Save owner priority and boost priority.
603  owner_prio_ = owner_->priority ();
604  if (prio_ceiling_ > owner_prio_)
605  {
606  boosted_prio_ = prio_ceiling_;
607 
608  // ----- Enter uncritical section -----------------------------------
609  scheduler::uncritical_section sucs;
610 
611  owner_->priority (boosted_prio_);
612  // ----- Exit uncritical section ------------------------------------
613  }
614  }
615 #if defined(OS_TRACE_RTOS_MUTEX)
616  trace::printf ("%s() @%p %s by %p %s LCK\n", __func__, this, name (),
617  crt_thread, crt_thread->name ());
618 #endif
619  return result::ok;
620  }
621 
622  // Relock.
623  if (saved_owner == crt_thread)
624  {
625  // The mutex was requested again by the same thread.
626  if (type_ == type::recursive)
627  {
628  if (count_ >= max_count_)
629  {
630  // The recursive mutex reached its limit.
631  return EAGAIN;
632  }
633 
634  // Increment the recursion depth counter.
635  ++count_;
636 
637 #if defined(OS_TRACE_RTOS_MUTEX)
638  trace::printf ("%s() @%p %s by %p %s >%u\n", __func__, this,
639  name (), crt_thread, crt_thread->name (), count_);
640 #endif
641  return result::ok;
642  }
643  else if (type_ == type::errorcheck)
644  {
645  // Recursive locks do not block, but return an error.
646  return EDEADLK;
647  }
648  else if (type_ == type::normal)
649  {
650 #if defined(OS_TRACE_RTOS_MUTEX)
651  trace::printf ("%s() @%p %s deadlock\n", __func__, this, name ());
652 #endif
653  return EWOULDBLOCK;
654  }
655  }
656  else
657  {
658  // Try to lock when not owner.
659 
660  // Another thread requested the mutex.
661  if (protocol_ == protocol::inherit)
662  {
663  thread::priority_t prio = crt_thread->priority ();
664  if ((prio > owner_->priority ()))
665  {
666  boosted_prio_ = prio;
667 
668  // ----- Enter uncritical section -----------------------------------
669  scheduler::uncritical_section sucs;
670 
671  // Boost owner priority.
672  owner_->priority (boosted_prio_);
673  // ----- Exit uncritical section ------------------------------------
674  }
675  }
676  }
677 
678  return EWOULDBLOCK;
679  }
