500  mutex::_try_lock (thread* crt_thread)
501  {
502  thread* saved_owner;
503 
504  saved_owner = owner_;
505  if (owner_ == nullptr)
506  {
507  owner_ = crt_thread;
508  count_ = 1;
509  ++(crt_thread->acquired_mutexes_);
510  }
511 
512  // ----- Enter uncritical section -----
513  scheduler::uncritical_section ucs;
514 
515  if (saved_owner == nullptr)
516  {
517  if (protocol_ == protocol::inherit)
518  {
519  // Save owner priority, in case a temporary boost
520  // will be later applied.
521  owner_prio_ = owner_->priority ();
522  }
523  else if (protocol_ == protocol::protect)
524  {
525  // Save owner priority and boost priority.
526  owner_prio_ = owner_->priority ();
527  if (prio_ceiling_ > owner_prio_)
528  {
529  owner_->priority (prio_ceiling_);
530  }
531  }
532 #if defined(OS_TRACE_RTOS_MUTEX)
533  trace::printf ("%s() @%p %s locked by %p %s\n", __func__, this,
534  name (), crt_thread, crt_thread->name ());
535 #endif
536  return result::ok;
537  }
538 
539  if (saved_owner == crt_thread)
540  {
541  if (type_ == type::recursive)
542  {
543  if (count_ == max_count_)
544  {
545  return EAGAIN;
546  }
547  ++count_;
548 #if defined(OS_TRACE_RTOS_MUTEX)
549  trace::printf ("%s() @%p %s incr %u by %p %s\n", __func__, this,
550  name (), count_, crt_thread, crt_thread->name ());
551 #endif
552  return result::ok;
553  }
554  else if (type_ == type::errorcheck)
555  {
556  return EDEADLK;
557  }
558  }
559 
560  if (protocol_ == protocol::inherit)
561  {
562  thread::priority_t prio = crt_thread->priority ();
563  if ((prio > owner_->priority ()))
564  {
565  // Boost owner priority.
566  owner_->priority (prio);
567  }
568  }
569 
570  return EWOULDBLOCK;
571 
572  // TODO: EINVAL, EOWNERDEAD
573  }
