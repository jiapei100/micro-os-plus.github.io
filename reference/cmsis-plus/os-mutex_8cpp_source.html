475  result_t
476  Mutex::_try_lock (Thread* crt_thread)
477  {
478  Thread* saved_owner;
479 
480  {
481  scheduler::Critical_section cs; // ----- Critical section -----
482 
483  saved_owner = owner_;
484  if (owner_ == nullptr)
485  {
486  owner_ = crt_thread;
487  count_ = 1;
488  }
489  // Short critical section ends here.
490  }
491 
492  if (saved_owner == nullptr)
493  {
494  if (protocol_ == mutex::protocol::inherit)
495  {
496  // Save owner priority, in case a temporary boost
497  // will be later applied.
498  owner_prio_ = owner_->sched_prio ();
499  }
500  else if (protocol_ == mutex::protocol::protect)
501  {
502  // Save owner priority and boost priority.
503  owner_prio_ = owner_->sched_prio ();
504  if (prio_ceiling_ > owner_prio_)
505  {
506  owner_->sched_prio (prio_ceiling_);
507  }
508  }
509  trace::printf ("mutex @%p %s locked by %p %s\n", this, name (),
510  crt_thread, crt_thread->name ());
511  return result::ok;
512  }
513 
514  if (saved_owner == crt_thread)
515  {
516  if (type_ == mutex::type::recursive)
517  {
518  if (count_ == max_count_)
519  {
520  return EAGAIN;
521  }
522  ++count_;
523  trace::printf ("mutex @%p %s incr %d by %p %s\n", this, name (),
524  count_, crt_thread, crt_thread->name ());
525  return result::ok;
526  }
527  else if (type_ == mutex::type::errorcheck)
528  {
529  return EDEADLK;
530  }
531  }
532 
533  if (protocol_ == mutex::protocol::inherit)
534  {
535  thread::priority_t prio = crt_thread->sched_prio ();
536  if ((prio > owner_->sched_prio ()))
537  {
538  // Boost owner priority.
539  owner_->sched_prio (prio);
540  }
541  }
542 
543  return EBUSY;
544 
545  // TODO: EINVAL, EOWNERDEAD
546  }

592  os_assert_err(!scheduler::in_handler_mode (), EPERM);
593 
594  trace::printf ("%s() @%p %s by %p %s\n", __func__, this, name (),
595  &this_thread::thread (), this_thread::thread ().name ());

604 
605  // Prepare a list node pointing to the current thread.
606  // Do not worry for being on stack, it is temporarily linked to the
607  // list and guaranteed to be removed before this function returns.
608  Double_list_node_thread node
609  { list_, crt_thread };
610 
611  for (;;)
612  {
613 
614  result_t res = _try_lock (&crt_thread);
615  if (res != EBUSY)
616  {
617  return res;
