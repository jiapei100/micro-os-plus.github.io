484  result_t
485  Mutex::_try_lock (Thread* crt_thread)
486  {
487  if (owner_ == nullptr)
488  {
489  owner_ = crt_thread;
490  count_ = 1;
491 
492  if (protocol_ == mutex::protocol::inherit)
493  {
494  // Save owner priority, in case a temporary boost
495  // will be later applied.
496  owner_prio_ = owner_->sched_prio ();
497  }
498  else if (protocol_ == mutex::protocol::protect)
499  {
500  // Save owner priority and boost priority.
501  owner_prio_ = owner_->sched_prio ();
502  if (prio_ceiling_ > owner_prio_)
503  {
504  owner_->sched_prio (prio_ceiling_);
505  }
506  }
507  trace::printf ("mutex @%p %s locked by %p %s\n", this, name (),
508  crt_thread, crt_thread->name ());
509  return result::ok;
510  }
511 
512  if (owner_ == crt_thread)
513  {
514  if (type_ == mutex::type::recursive)
515  {
516  if (count_ == max_count_)
517  {
518  return EAGAIN;
519  }
520  ++count_;
521  trace::printf ("mutex @%p %s incr %d by %p %s\n", this, name (),
522  count_, crt_thread, crt_thread->name ());
523  return result::ok;
524  }
525  else if (type_ == mutex::type::errorcheck)
526  {
527  return EDEADLK;
528  }
529  }
530 
531  if (protocol_ == mutex::protocol::inherit)
532  {
533  thread::priority_t prio = crt_thread->sched_prio ();
534  if ((prio > owner_->sched_prio ()))
535  {
536  // Boost owner priority.
537  owner_->sched_prio (prio);
538  }
539  }
540 
541  return EBUSY;
542 
543  // TODO: EINVAL, EOWNERDEAD
544  }
