492  result_t
493  Mutex::_try_lock (Thread* crt_thread)
494  {
495  Thread* saved_owner;
496 
497  saved_owner = owner_;
498  if (owner_ == nullptr)
499  {
500  owner_ = crt_thread;
501  count_ = 1;
502  ++(crt_thread->acquired_mutexes_);
503  }
504 
505  if (saved_owner == nullptr)
506  {
507  if (protocol_ == mutex::protocol::inherit)
508  {
509  // Save owner priority, in case a temporary boost
510  // will be later applied.
511  owner_prio_ = owner_->sched_prio ();
512  }
513  else if (protocol_ == mutex::protocol::protect)
514  {
515  // Save owner priority and boost priority.
516  owner_prio_ = owner_->sched_prio ();
517  if (prio_ceiling_ > owner_prio_)
518  {
519  owner_->sched_prio (prio_ceiling_);
520  }
521  }
522 #if defined(OS_TRACE_RTOS_MUTEX)
523  trace::printf ("mutex @%p %s locked by %p %s\n", this, name (),
524  crt_thread, crt_thread->name ());
525 #endif
526  return result::ok;
527  }
528 
529  if (saved_owner == crt_thread)
530  {
531  if (type_ == mutex::type::recursive)
532  {
533  if (count_ == max_count_)
534  {
535  return EAGAIN;
536  }
537  ++count_;
538 #if defined(OS_TRACE_RTOS_MUTEX)
539  trace::printf ("mutex @%p %s incr %d by %p %s\n", this, name (),
540  count_, crt_thread, crt_thread->name ());
541 #endif
542  return result::ok;
543  }
544  else if (type_ == mutex::type::errorcheck)
545  {
546  return EDEADLK;
547  }
548  }
549 
550  if (protocol_ == mutex::protocol::inherit)
551  {
552  thread::priority_t prio = crt_thread->sched_prio ();
553  if ((prio > owner_->sched_prio ()))
554  {
555  // Boost owner priority.
556  owner_->sched_prio (prio);
557  }
558  }
559 
560  return EBUSY;
561 
562  // TODO: EINVAL, EOWNERDEAD
563  }
