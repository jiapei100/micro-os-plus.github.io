459  void
460  memory_resource::internal_increase_allocated_statistics (
461  std::size_t bytes) noexcept
462  {
463  // Update statistics.
464  // What is subtracted from free is added to allocated.
465  allocated_bytes_ += bytes;
466  if (allocated_bytes_ > max_allocated_bytes_)
467  {
468  max_allocated_bytes_ = allocated_bytes_;
469  }
470  free_bytes_ -= bytes;
471  ++allocated_chunks_;
472  --free_chunks_;
473  }
474 
475  void
476  memory_resource::internal_decrease_allocated_statistics (
477  std::size_t bytes) noexcept
478  {
479  // Update statistics.
480  // What is subtracted from allocated is added to free.
481  allocated_bytes_ -= bytes;
482  free_bytes_ += bytes;
483  --allocated_chunks_;
484  ++free_chunks_;
485 
486  }
