191  {
192  os_assert_err(!interrupts::in_handler_mode (), EPERM);
193  os_assert_err(!scheduler::locked (), EPERM);
194

196  trace::printf ("%s(%u)\n", __func__,
197  static_cast<unsigned int> (duration));
198 #endif
199 
200  clock::timestamp_t timestamp = steady_now () + duration;
201  for (;;)
202  {
203  result_t res;

205 
206  timestamp_t n = steady_now ();

208  {

314  result_t
315  clock::_wait_until (timestamp_t timestamp, clock_timestamps_list& list)
316  {
317  thread& crt_thread = this_thread::thread ();
318 
319  // Prepare a list node pointing to the current thread.
320  // Do not worry for being on stack, it is temporarily linked to the
321  // list and guaranteed to be removed before this function returns.
322  timeout_thread_node node
323  { timestamp, crt_thread };
324 
325  {
326  interrupts::critical_section ics; // ----- Critical section -----
327 
328  // Remove this thread from the ready list, if there.
329  port::this_thread::prepare_suspend ();
330 
331  // Add this thread to the clock waiting list.
332  list.link (node);
333  crt_thread.clock_node_ = &node;
334  crt_thread.sched_state_ = thread::state::suspended;
335  }
336 
337  port::scheduler::reschedule ();
338 
339  {
340  interrupts::critical_section ics; // ----- Critical section -----
341 
342  // Remove the thread from the clock timeout list,
343  // if not already removed by the timer.
344  crt_thread.clock_node_ = nullptr;
345  node.unlink ();
346  }
347 
348  return result::ok;
349  }
