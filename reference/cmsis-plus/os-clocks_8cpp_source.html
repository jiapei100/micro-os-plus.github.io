351  clock::timestamp_t
352  Systick_clock::now (current_t* details)
353  {
354  assert(details != nullptr);
355 
356  // The core frequency can be returned right away, since
357  // is not expected to change during this call.
358  details->core_frequency_hz = SystemCoreClock;
359 
360  // The timer reload value, it is the divisor - 1.
361  uint32_t load = SysTick->LOAD;
362  // Can also be returned right away.
363  details->divisor = load + 1;
364 
365  // The ticks and cycles must be read atomically, use a critical
366  // section and adjust for border condition, when the event
367  // came exactly after entering the critical section.
368  uint64_t ticks;
369  uint32_t val;
370  {
371  interrupts::Critical_section cs; // ----- Critical section -----
372 
373  // Sample ticks counter inside critical section.
374  ticks = steady_count_;
375 
376  // Initial sample of the decrementing counter.
377  // Might happen before the event, will be used as such.
378  val = SysTick->VAL;
379 
380  // Check overflow. If the exception is pending, it means the
381  // ticks counter was not yet updated, but the counter was reloaded.
382  if (SysTick->CTRL & SCB_ICSR_PENDSTSET_Msk)
383  {
384  // Sample the decrementing counter again to validate the
385  // initial sample.
386  uint32_t val_update = SysTick->VAL;
387  if (val_update > val)
388  {
389  // The second sample is more accurate.
390  val = val_update;
391  }
392  ticks++; // Adjust to next tick.
393  }
394  }
395  details->cycles = load - val;
396  details->ticks = ticks;
397 
398  return ticks;
399  }
