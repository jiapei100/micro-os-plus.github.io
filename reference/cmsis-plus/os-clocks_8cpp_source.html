293  clock::timestamp_t
294  Systick_clock::now (current_t* details)
295  {
296  assert(details != nullptr);
297 
298  // The core frequency can be returned right away, since
299  // is not expected to change during this call.
300  details->core_frequency_hz = SystemCoreClock;
301 
302  // The timer reload value, it is the divisor - 1.
303  uint32_t load = SysTick->LOAD;
304  // Can also be returned right away.
305  details->divisor = load + 1;
306 
307  // The ticks and cycles must be read atomically, use a critical
308  // section and adjust for border condition, when the event
309  // came exactly after entering the critical section.
310  uint64_t ticks;
311  uint32_t val;
312  {
313  interrupts::Critical_section cs; // ----- Critical section -----
314 
315  // Sample ticks counter inside critical section.
316  ticks = steady_count_;
317 
318  // Initial sample of the decrementing counter.
319  // Might happen before the event, will be used as such.
320  val = SysTick->VAL;
321 
322  // Check overflow. If the exception is pending, it means the
323  // ticks counter was not yet updated, but the counter was reloaded.
324  if (SysTick->CTRL & SCB_ICSR_PENDSTSET_Msk)
325  {
326  // Sample the decrementing counter again to validate the
327  // initial sample.
328  uint32_t val_update = SysTick->VAL;
329  if (val_update > val)
330  {
331  // The second sample is more accurate.
332  val = val_update;
333  }
334  ticks++; // Adjust to next tick.
335  }
336  }
337  details->cycles = load - val;
338  details->ticks = ticks;
339 
340  return ticks;
341  }
