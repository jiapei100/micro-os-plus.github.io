240  template<typename T>
241  constexpr std::size_t
242  compute_allocated_size_bytes (std::size_t msgs,
243  std::size_t msg_size_bytes)
244  {
245  // Align each message
246  return (msgs * ((msg_size_bytes + (sizeof(T) - 1)) & ~(sizeof(T) - 1)))
247  // Align the indices array
248  + ((2 * msgs * sizeof(index_t) + (sizeof(T) - 1))
249  & ~(sizeof(T) - 1))
250  // Align the priority array
251  + ((msgs * sizeof(priority_t) + (sizeof(T) - 1))
252  & ~(sizeof(T) - 1));
253  }
