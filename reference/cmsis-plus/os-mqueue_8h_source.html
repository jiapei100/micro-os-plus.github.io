130  // Storage for a message queue. Each message is stored in an element
131  // extended to a multiple of pointers. The lists are kept in two arrays
132  // of indices and the priorities in a separate array.
133  template<typename T, std::size_t msgs, std::size_t msg_size_bytes>
134  class Arena
135  {
136  public:
137  T queue[(msgs * msg_size_bytes + sizeof(T) - 1) / sizeof(T)];
138  T links[((2 * msgs) * sizeof(mqueue::index_t) + sizeof(T) - 1)
139  / sizeof(T)];
140  T prios[(msgs * sizeof(mqueue::priority_t) + sizeof(T) - 1)
141  / sizeof(T)];
142  };
143 
144  template<typename T>
145  constexpr std::size_t
146  compute_allocated_size_bytes (std::size_t msgs,
147  std::size_t msg_size_bytes)
148  {
149  // Align each message
150  return (msgs * ((msg_size_bytes + (sizeof(T) - 1)) & ~(sizeof(T) - 1)))
151  // Align the indices array
152  + ((2 * msgs * sizeof(mqueue::index_t) + (sizeof(T) - 1))
153  & ~(sizeof(T) - 1))
154  // Align the priority array
155  + ((msgs * sizeof(mqueue::priority_t) + (sizeof(T) - 1))
156  & ~(sizeof(T) - 1));
157  }
