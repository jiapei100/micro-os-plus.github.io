244  template<typename T>
245  constexpr std::size_t
246  compute_allocated_size_bytes (std::size_t msgs,
247  std::size_t msg_size_bytes)
248  {
249  // Align each message
250  return (msgs * ((msg_size_bytes + (sizeof(T) - 1)) & ~(sizeof(T) - 1)))
251  // Align the indices array
252  + ((2 * msgs * sizeof(index_t) + (sizeof(T) - 1))
253  & ~(sizeof(T) - 1))
254  // Align the priority array
255  + ((msgs * sizeof(priority_t) + (sizeof(T) - 1))
256  & ~(sizeof(T) - 1));
257  }

1412  template<typename Allocator>
1413  message_queue_allocated<Allocator>::message_queue_allocated (
1414  const char* name, std::size_t msgs, std::size_t msg_size_bytes,
1415  const attributes& attr, const allocator_type& allocator) :
1416  message_queue
1417  { name }
1418  {
1419 #if defined(OS_TRACE_RTOS_MQUEUE)
1420  trace::printf ("%s() @%p %s %d %d\n", __func__, this, this->name (),
1421  msgs, msg_size_bytes);
1422 #endif
1423 
1424  if (attr.mq_queue_address != nullptr)
1425  {
1426  // Do not use any allocator at all.
1427  internal_construct_ (msgs, msg_size_bytes, attr, nullptr, 0);
1428  }
1429  else
1430  {
1431  allocator_ = &allocator;
1432 
1433  // If no user storage was provided via attributes,
1434  // allocate it dynamically via the allocator.
1435  allocated_queue_size_elements_ = (compute_allocated_size_bytes<
1436  typename allocator_type::value_type> (msgs, msg_size_bytes)
1437  + sizeof(typename allocator_type::value_type) - 1)
1438  / sizeof(typename allocator_type::value_type);
1439 
1440  allocated_queue_addr_ =
1441  const_cast<allocator_type&> (allocator).allocate (
1442  allocated_queue_size_elements_);
1443 
1444  internal_construct_ (
1445  msgs,
1446  msg_size_bytes,
1447  attr,
1448  allocated_queue_addr_,
1449  allocated_queue_size_elements_
1450  * sizeof(typename allocator_type::value_type));
1451  }
1452  }
