251  template<typename T>
252  constexpr std::size_t
253  compute_allocated_size_bytes (std::size_t msgs,
254  std::size_t msg_size_bytes)
255  {
256  // Align each message
257  return (msgs * ((msg_size_bytes + (sizeof(T) - 1)) & ~(sizeof(T) - 1)))
258  // Align the indices array
259  + ((2 * msgs * sizeof(index_t) + (sizeof(T) - 1))
260  & ~(sizeof(T) - 1))
261  // Align the priority array
262  + ((msgs * sizeof(priority_t) + (sizeof(T) - 1))
263  & ~(sizeof(T) - 1));
264  }

1428  // allocate it dynamically via the allocator.
1429  allocated_queue_size_elements_ = (compute_allocated_size_bytes<
1430  typename allocator_type::value_type> (msgs, msg_size_bytes)

1432  / sizeof(typename allocator_type::value_type);

1435  const_cast<allocator_type&> (allocator).allocate (
1436  allocated_queue_size_elements_);

1439  msgs,
1440  msg_size_bytes,
