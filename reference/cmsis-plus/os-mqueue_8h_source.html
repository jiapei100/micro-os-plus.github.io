130  // Storage for a message queue. Each message is stored in an element
131  // extended to a multiple of pointers. The lists are kept in two arrays
132  // of indices and the priorities in a separate array.
133  template<typename T, std::size_t msgs, std::size_t msg_size_bytes>
134  class Arena
135  {
136  public:
137  T queue[msgs * msg_size_bytes / sizeof(T)];
138  T links[(2 * msgs) * sizeof(mqueue::index_t) / sizeof(T)];
139  T prios[msgs * sizeof(mqueue::priority_t) / sizeof(T)];
140  };
141 
142  template<typename T>
143  constexpr std::size_t
144  compute_allocated_size_bytes (std::size_t msgs,
145  std::size_t msg_size_bytes)
146  {
147  // Align each message
148  return (msgs * ((msg_size_bytes + (sizeof(T) - 1)) & ~(sizeof(T) - 1)))
149  // Align the indices array
150  + ((2 * msgs * sizeof(mqueue::index_t) + (sizeof(T) - 1))
151  & ~(sizeof(T) - 1))
152  // Align the priority array
153  + ((msgs * sizeof(mqueue::priority_t) + (sizeof(T) - 1))
154  & ~(sizeof(T) - 1));
155  }
