213  bool
214  Event_flags::_try_wait (flags::mask_t mask, flags::mask_t* oflags,
215  flags::mode_t mode)
216  {
217  if ((mask != 0) && ((mode & flags::mode::all) != 0))
218  {
219  // Only if all desires signals are raised we're done.
220  if ((flags_ & mask) == mask)
221  {
222  if (oflags != nullptr)
223  {
224  *oflags = flags_;
225  }
226  // Clear desired signals.
227  flags_ &= ~mask;
228  return true;
229  }
230  }
231  else if ((mask == 0) || ((mode & flags::mode::any) != 0))
232  {
233  // Any flag will do it.
234  if (flags_ != 0)
235  {
236  // Possibly return.
237  if (oflags != nullptr)
238  {
239  *oflags = flags_;
240  }
241  // Since we returned them all, also clear them all.
242  flags_ = 0;
243  return true;
244  }
245  }
246 
247  return false;
248  }
