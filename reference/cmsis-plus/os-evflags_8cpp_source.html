223  bool
224  event_flags::_try_wait (flags::mask_t mask, flags::mask_t* oflags,
225  flags::mode_t mode)
226  {
227  if ((mask != 0) && ((mode & flags::mode::all) != 0))
228  {
229  // Only if all desires signals are raised we're done.
230  if ((flags_ & mask) == mask)
231  {
232  if (oflags != nullptr)
233  {
234  *oflags = flags_;
235  }
236  // Clear desired signals.
237  flags_ &= ~mask;
238  return true;
239  }
240  }
241  else if ((mask == 0) || ((mode & flags::mode::any) != 0))
242  {
243  // Any flag will do it.
244  if (flags_ != 0)
245  {
246  // Possibly return.
247  if (oflags != nullptr)
248  {
249  *oflags = flags_;
250  }
251  // Since we returned them all, also clear them all.
252  flags_ = 0;
253  return true;
254  }
255  }
256 
257  return false;
258  }

304

306  {

436  event_flags::timed_wait (flags::mask_t mask, clock::duration_t timeout,
437  flags::mask_t* oflags, flags::mode_t mode)
438  {
439  os_assert_throw(!interrupts::in_handler_mode (), EPERM);

442 #if defined(OS_TRACE_RTOS_EVFLAGS)

444 #endif
445 
446 #if defined(OS_USE_RTOS_PORT_EVENT_FLAGS)
447 
448  return port::event_flags::timed_wait (this, mask, timeout, oflags, mode);
449 
450 #else
451 
452  // Extra test before entering the loop, with its inherent weight.
453  // Trade size for speed.
454  {
455  interrupts::critical_section ics; // ----- Critical section -----
456 
457  if (_try_wait (mask, oflags, mode))
458  {
459  return result::ok;
460  }
461  }
462 
463  thread& crt_thread = this_thread::thread ();
464 
465  // Prepare a list node pointing to the current thread.
466  // Do not worry for being on stack, it is temporarily linked to the
467  // list and guaranteed to be removed before this function returns.
468  waiting_thread_node node
469  { crt_thread };
470 
471  clock_timestamps_list& clock_list = clock_->steady_list ();
472  clock::timestamp_t timeout_timestamp = clock_->steady_now () + timeout;
473 
474  // Prepare a timeout node pointing to the current thread.
475  timeout_thread_node timeout_node
476  { timeout_timestamp, crt_thread };
477 
478  for (;;)
479  {
480  {
481  interrupts::critical_section ics; // ----- Critical section -----
482 
483  if (_try_wait (mask, oflags, mode))
484  {
485  return result::ok;
486  }
487 
488  // Add this thread to the event flags waiting list,
489  // and the clock timeout list.
490  scheduler::_link_node (list_, node, clock_list, timeout_node);
491  // state::suspended set in above link().
492  }
493 
494  port::scheduler::reschedule ();
495 
496  // Remove the thread from the event flags waiting list,
497  // if not already removed by raise() and from the clock
498  // timeout list, if not already removed by the timer.
499  scheduler::_unlink_node (node, timeout_node);
500 
501  if (crt_thread.interrupted ())
502  {
503  return EINTR;
504  }
505 
506  if (clock_->steady_now () >= timeout_timestamp)
507  {
508  return ETIMEDOUT;
509  }
510  }
511 
512  /* NOTREACHED */
513  return ENOTRECOVERABLE;
514 
515 #endif
