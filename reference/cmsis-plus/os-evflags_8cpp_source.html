213  bool
214  Event_flags::_try_wait (flags::mask_t mask, flags::mask_t* oflags,
215  flags::mode_t mode)
216  {
217  interrupts::Critical_section cs; // ----- Critical section -----
218 
219  if ((mask != 0) && ((mode & flags::mode::all) != 0))
220  {
221  // Only if all desires signals are raised we're done.
222  if ((flags_ & mask) == mask)
223  {
224  if (oflags != nullptr)
225  {
226  *oflags = flags_;
227  }
228  // Clear desired signals.
229  flags_ &= ~mask;
230  return true;
231  }
232  }
233  else if ((mask == 0) || ((mode & flags::mode::any) != 0))
234  {
235  // Any flag will do it.
236  if (flags_ != 0)
237  {
238  // Possibly return.
239  if (oflags != nullptr)
240  {
241  *oflags = flags_;
242  }
243  // Since we returned them all, also clear them all.
244  flags_ = 0;
245  return true;
246  }
247  }
248 
249  return false;
250  }

391  os_assert_throw(!scheduler::in_handler_mode (), EPERM);
392 
393  if (timeout == 0)
394  {
395  timeout = 1;
396  }
397 
398 #if defined(OS_INCLUDE_RTOS_PORT_EVENT_FLAGS)
399 
400  return port::Event_flags::timed_wait (this, mask, timeout, oflags, mode);
401 
402 #else
403 
404  Thread& crt_thread = this_thread::thread ();
405 
406  // Prepare a list node pointing to the current thread.
407  // Do not worry for being on stack, it is temporarily linked to the
408  // list and guaranteed to be removed before this function returns.
409  Waiting_thread_node node
410  { list_, crt_thread };
411 
412  clock::timestamp_t start = clock_.steady_now ();
413  for (;;)
414  {
415  clock::duration_t spent;
416 
417  if (_try_wait (mask, oflags, mode))
418  {
419  return result::ok;
420  }
421 
422  clock::timestamp_t now = clock_.steady_now ();
423  spent = (clock::duration_t) (now - start);
424  if (spent >= timeout)
425  {
426  return ETIMEDOUT;
427  }
428 
429  {
430  // Add this thread to the event flags waiting list.
431  // It is removed when this block ends (after wait_for()).
432  Waiting_threads_list_guard<interrupts::Critical_section> lg
433  { node };
434 
435  clock_.wait_for (timeout - spent);
436  }
437 
438  if (crt_thread.interrupted ())
439  {
440  return EINTR;
441  }
442  }
443
