60  ssize_t
61  write (const void* buf, std::size_t nbyte)
62  {
63  if (buf == nullptr || nbyte == 0)
64  {
65  return 0;
66  }
67 
68  const char* cbuf = (const char*) buf;
69 
70  for (size_t i = 0; i < nbyte; i++)
71  {
72  // Check if ITM or the stimulus port are not enabled.
73  if (((ITM->TCR & ITM_TCR_ITMENA_Msk) == 0)
74  || ((ITM->TER & (1UL << OS_INTEGER_TRACE_ITM_STIMULUS_PORT))
75  == 0))
76  {
77  // Return the number of sent characters (may be 0).
78  return (ssize_t) i;
79  }
80 
81  // Wait until STIMx is ready...
82  while (ITM->PORT[OS_INTEGER_TRACE_ITM_STIMULUS_PORT].u32 == 0)
83  ;
84  // then send data, one byte at a time
85  ITM->PORT[OS_INTEGER_TRACE_ITM_STIMULUS_PORT].u8 =
86  (uint8_t) (*cbuf++);
87  }
88 
89  // All characters successfully sent.
90  return (ssize_t) nbyte;
91  }
