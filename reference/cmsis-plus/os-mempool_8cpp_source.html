299  * Construct the linked list of blocks and initialise the
300  * internal pointers and counters.

302  void
303  Memory_pool::_init (void)
304  {
305  // Construct a linked list of blocks. Store the pointer at
306  // the beginning of each block. Each block
307  // will hold the address of the next free block, or nullptr at the end.
308  char* p = pool_addr_;
309  for (std::size_t i = 1; i < blocks_; ++i)
310  {
311  *(void**) p = (p + block_size_bytes_);
312  p += block_size_bytes_;
313  }
314  *(void**) p = nullptr;
315 
316  first_ = pool_addr_; // Pointer to first block.
317 
318  count_ = 0; // No allocated blocks.
319  }
