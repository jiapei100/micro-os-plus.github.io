199  Memory_pool::Memory_pool (const mempool::Attributes& attr,
200  mempool::size_t blocks,
201  mempool::size_t block_size_bytes) :
202  Named_object
203  { attr.name () }, //
204  blocks_ (blocks), //
205  block_size_bytes_ (block_size_bytes)
206 
207  {
208  os_assert_throw(!scheduler::in_handler_mode (), EPERM);
209 
210  pool_addr_ = (char*) attr.mp_pool_address;
211 
212  assert(blocks_ > 0);
213  // Blocks must be large enough to store the index, used
214  // to construct the list of free blocks.
215  assert(block_size_bytes_ >= sizeof(std::ptrdiff_t));
216 
217  flags_ = 0;
218 
219  trace::printf ("%s() @%p %s %d %d\n", __func__, this, name (), blocks_,
220  block_size_bytes_);
221 
222  if (pool_addr_ == nullptr)
223  {
224  pool_addr_ = new (std::nothrow) char[blocks_ * block_size_bytes_];
225  flags_ |= flags_allocated;
226  }
227  else
228  {
229  os_assert_throw(
230  attr.mp_pool_size_bytes
231  >= ((std::size_t )(blocks_ * block_size_bytes_)),
232  ENOMEM);
233  }
234 
235  os_assert_throw(pool_addr_ != nullptr, ENOMEM);
236 
237  // Construct a linked list of blocks. Store the pointer at
238  // the beginning of each block. Each block
239  // will hold the address of the next free block, or nullptr at the end.
240  char* p = pool_addr_;
241  for (std::size_t i = 1; i < blocks_; ++i)
242  {
243  *(void**) p = (p + block_size_bytes_);
244  p += block_size_bytes_;
245  }
246  *(void**) p = nullptr;
247 
248  first_ = pool_addr_; // Pointer to first block.
249 
250  count_ = 0; // No allocated blocks.
251  }

438  Memory_pool::free (void* block)
439  {
440  trace::printf ("%s() @%p %s\n", __func__, this, name ());
441 
442  // Validate pointer.
443  if ((block < pool_addr_)
444  || (block >= (pool_addr_ + blocks_ * block_size_bytes_)))
445  {
446  return EINVAL;
447  }
448 
449  interrupts::Critical_section cs; // ----- Critical section -----
450 
451  // Perform a push_front() on the single linked LIFO list,
452  // i.e. add the block to the beginning of the list.
453 
454  // Link previous list to this block; may be null, but it does
455  // not matter.
456  *(void**) block = first_;
457 
458  // Now this block is the first one.
459  first_ = block;
460 
461  --count_;
462 
463  if (!list_.empty ())
464  {
465  // Wake-up one thread, if any.
466  list_.wakeup_one ();
467  }
468 
469  return result::ok;
470  }
