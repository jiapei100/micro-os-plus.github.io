312  void
313  Memory_pool::_init (void)
314  {
315  // Construct a linked list of blocks. Store the pointer at
316  // the beginning of each block. Each block
317  // will hold the address of the next free block, or nullptr at the end.
318  char* p = pool_addr_;
319  for (std::size_t i = 1; i < blocks_; ++i)
320  {
321  *(void**) p = (p + block_size_bytes_);
322  p += block_size_bytes_;
323  }
324  *(void**) p = nullptr;
325 
326  first_ = pool_addr_; // Pointer to first block.
327 
328  count_ = 0; // No allocated blocks.
329  }

504  result_t
505  Memory_pool::free (void* block)
506  {
507  trace::printf ("%s() @%p %s\n", __func__, this, name ());
508 
509  // Validate pointer.
510  if ((block < pool_addr_)
511  || (block >= (pool_addr_ + blocks_ * block_size_bytes_)))
512  {
513  return EINVAL;
514  }
515 
516  {
517  interrupts::Critical_section cs; // ----- Critical section -----
518 
519  // Perform a push_front() on the single linked LIFO list,
520  // i.e. add the block to the beginning of the list.
521 
522  // Link previous list to this block; may be null, but it does
523  // not matter.
524  *(void**) block = first_;
525 
526  // Now this block is the first one.
527  first_ = block;
528 
529  --count_;
530  }
531 
532  // Wake-up one thread, if any.
533  list_.resume_one ();
534 
535  return result::ok;
536  }
