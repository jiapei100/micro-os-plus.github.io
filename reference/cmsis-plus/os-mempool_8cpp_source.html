325  /*
326  * Construct the linked list of blocks and initialise the
327  * internal pointers and counters.
328  */
329  void
330  Memory_pool::_init (void)
331  {
332  // Construct a linked list of blocks. Store the pointer at
333  // the beginning of each block. Each block
334  // will hold the address of the next free block, or nullptr at the end.
335  char* p = static_cast<char*> (pool_addr_);
336  for (std::size_t i = 1; i < blocks_; ++i)
337  {
338  // Compute the address of the next block;
339  char* pn = p + block_size_bytes_;
340 
341  // Make this block point to the next one.
342  *(static_cast<void**> (static_cast<void*> (p))) = pn;
343  // Advance pointer
344  p = pn;
345  }
346 
347  // Mark end of list.
348  *(static_cast<void**> (static_cast<void*> (p))) = nullptr;
349 
350  first_ = pool_addr_; // Pointer to first block.
351 
352  count_ = 0; // No allocated blocks.
353  }

610  result_t
611  Memory_pool::free (void* block)
612  {
613 #if defined(OS_TRACE_RTOS_MEMPOOL)
614  trace::printf ("%s() @%p %s\n", __func__, this, name ());
615 #endif
616 
617  // Validate pointer.
618  if ((block < pool_addr_)
619  || (block
620  >= (static_cast<char*> (pool_addr_) + blocks_ * block_size_bytes_)))
621  {
622  return EINVAL;
623  }
624 
625  {
626  interrupts::Critical_section ics; // ----- Critical section -----
627 
628  // Perform a push_front() on the single linked LIFO list,
629  // i.e. add the block to the beginning of the list.
630 
631  // Link previous list to this block; may be null, but it does
632  // not matter.
633  *(static_cast<void**> (block)) = first_;
634 
635  // Now this block is the first one.
636  first_ = block;
637 
638  --count_;
639  }
640 
641  // Wake-up one thread, if any.
642  list_.resume_one ();
643 
644  return result::ok;
645  }
