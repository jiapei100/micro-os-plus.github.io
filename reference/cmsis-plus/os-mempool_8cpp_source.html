312  void
313  memory_pool::_construct (std::size_t blocks, std::size_t block_size_bytes,
314  const attributes& attr, void* pool_address,
315  std::size_t pool_size_bytes)
316  {
317  os_assert_throw(!interrupts::in_handler_mode (), EPERM);
318 
319 #if !defined(OS_USE_RTOS_PORT_MEMORY_POOL)
320  clock_ = attr.clock != nullptr ? attr.clock : &sysclock;
321 #endif
322 
323  blocks_ = static_cast<memory_pool::size_t> (blocks);
324  assert(blocks_ == blocks);
325  assert(blocks_ > 0);
326 
327  // Adjust block size to multiple of pointer.
328  // Blocks must be large enough to store a pointer, used
329  // to construct the list of free blocks.
330  block_size_bytes_ = (static_cast<memory_pool::size_t> (block_size_bytes
331  + __SIZEOF_POINTER__ - 1))
332  & (static_cast<memory_pool::size_t> (~(__SIZEOF_POINTER__ - 1)));
333 
334  // If the storage is given explicitly, override attributes.
335  if (pool_address != nullptr)
336  {
337  // The attributes should not define any storage in this case.
338  assert(attr.mp_pool_address == nullptr);
339 
340  pool_addr_ = pool_address;
341  pool_size_bytes_ = pool_size_bytes;
342  }
343  else
344  {
345  pool_addr_ = attr.mp_pool_address;
346  pool_size_bytes_ = attr.mp_pool_size_bytes;
347  }
348 
349  // Blocks must be pointer aligned.
350  void* p = pool_addr_;
351  std::size_t sz = pool_size_bytes_;
352  pool_addr_ = static_cast<char*> (std::align (__SIZEOF_POINTER__,
353  blocks_ * block_size_bytes_,
354  p, sz));
355 
356 #if defined(OS_TRACE_RTOS_MEMPOOL)
357  trace::printf ("%s() @%p %s %u %u %p %u\n", __func__, this, name (),
358  blocks_, block_size_bytes_, pool_addr_, pool_size_bytes_);
359 #endif
360 
361  std::size_t storage_size = compute_allocated_size_bytes<void*> (
362  blocks_, block_size_bytes_);
363 
364  if (pool_addr_ != nullptr)
365  {
366  os_assert_throw(pool_size_bytes_ > 0, EINVAL);
367  os_assert_throw(pool_size_bytes_ >= storage_size, EINVAL);
368  }
369 
370  os_assert_throw(pool_addr_ != nullptr, ENOMEM);
371 
372  _init ();
373 
374  }

418  /*
419  * Construct the linked list of blocks and initialise the
420  * internal pointers and counters.
421  */
422  void
423  memory_pool::_init (void)
424  {
425  // Construct a linked list of blocks. Store the pointer at
426  // the beginning of each block. Each block
427  // will hold the address of the next free block, or nullptr at the end.
428  char* p = static_cast<char*> (pool_addr_);
429  for (std::size_t i = 1; i < blocks_; ++i)
430  {
431  // Compute the address of the next block;
432  char* pn = p + block_size_bytes_;
433 
434  // Make this block point to the next one.
435  *(static_cast<void**> (static_cast<void*> (p))) = pn;
436  // Advance pointer
437  p = pn;
438  }
439 
440  // Mark end of list.
441  *(static_cast<void**> (static_cast<void*> (p))) = nullptr;
442 
443  first_ = pool_addr_; // Pointer to first block.
444 
445  count_ = 0; // No allocated blocks.
446  }

712  result_t
713  memory_pool::free (void* block)
714  {
715 #if defined(OS_TRACE_RTOS_MEMPOOL)
716  trace::printf ("%s() @%p %s\n", __func__, this, name ());
717 #endif
718 
719  assert(port::interrupts::is_priority_valid ());
720 
721  // Validate pointer.
722  if ((block < pool_addr_)
723  || (block
724  >= (static_cast<char*> (pool_addr_) + blocks_ * block_size_bytes_)))
725  {
726  return EINVAL;
727  }
728 
729  {
730  interrupts::critical_section ics; // ----- Critical section -----
731 
732  // Perform a push_front() on the single linked LIFO list,
733  // i.e. add the block to the beginning of the list.
734 
735  // Link previous list to this block; may be null, but it does
736  // not matter.
737  *(static_cast<void**> (block)) = first_;
738 
739  // Now this block is the first one.
740  first_ = block;
741 
742  --count_;
743  }
744 
745  // Wake-up one thread, if any.
746  list_.resume_one ();
747 
748  return result::ok;
749  }
