74  io*
75  vopen (const char* path, int oflag, std::va_list args)
76  {
77  if (path == nullptr)
78  {
79  errno = EFAULT;
80  return nullptr;
81  }
82 
83  if (*path == '\0')
84  {
85  errno = ENOENT;
86  return nullptr;
87  }
88 
89  errno = 0;
90 
91  // First check if path is a char device.
92  os::posix::io* io = os::posix::device_char_registry::identify_device (
93  path);
94  if (io != nullptr)
95  {
96  // If so, use the implementation to open the device.
97  int oret = static_cast<device_char*> (io)->do_vopen (path, oflag,
98  args);
99  if (oret < 0)
100  {
101  // Open failed.
102  return nullptr;
103  }
104  }
105  else
106  {
107  auto adjusted_path = path;
108  auto* const fs = os::posix::mount_manager::identify_file_system (
109  &adjusted_path);
110 
111  // The manager will return null if there are no file systems
112  // registered, no need to check this condition separately.
113  if (fs == nullptr)
114  {
115  errno = EBADF;
116  return nullptr;
117  }
118 
119  // Use the file system implementation to open the file, using
120  // the adjusted path (mount point prefix removed).
121  io = fs->open (adjusted_path, oflag, args);
122  if (io == nullptr)
123  {
124  // Open failed.
125  return nullptr;
126  }
127  }
128 
129  // If successful, allocate a file descriptor.
130  // Return a valid pointer to an object derived from io, or nullptr.
131 
132  auto ret = io->alloc_file_descriptor ();
133  trace::printf ("%s(\"%s\")=%p fd=%d\n", __func__, path, io,
134  io->file_descriptor ());
135  return ret;
136  }
