109  Event_flags::_try_wait (flags::mask_t mask, flags::mask_t* oflags,
110  flags::mode_t mode)
111  {
112  if ((mask != 0) && ((mode & flags::mode::all) != 0))
113  {
114  // Only if all desires signals are raised we're done.
115  if ((flags_ & mask) == mask)
116  {
117  if (oflags != nullptr)
118  {
119  *oflags = flags_;
120  }
121  // Clear desired signals.
122  flags_ &= ~mask;
123  return result::ok;
124  }
125  }
126  else if ((mask == 0) || ((mode & flags::mode::any) != 0))
127  {
128  // Any flag will do it.
129  if (flags_ != 0)
130  {
131  // Possibly return.
132  if (oflags != nullptr)
133  {
134  *oflags = flags_;
135  }
136  // Since we returned them all, also clear them all.
137  flags_ = 0;
138  return result::ok;
139  }
140  }
141 
142  return EAGAIN;
143  }
