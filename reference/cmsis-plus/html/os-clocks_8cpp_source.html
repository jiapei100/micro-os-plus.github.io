132  Systick_clock::rep
133  Systick_clock::now (current_t* details)
134  {
135  assert(details != nullptr);
136 
137  // The core frequency can be returned right away, since
138  // is not expected to change during this call.
139  details->core_frequency_hz = SystemCoreClock;
140 
141  // The timer reload value, it is the divisor - 1.
142  uint32_t load = SysTick->LOAD;
143  // Can also be returned right away.
144  details->divisor = load + 1;
145 
146  // The ticks and cycles must be read atomically, use a critical
147  // section and adjust for border condition, when the event
148  // came exactly after entering the critical section.
149  uint64_t ticks;
150  uint32_t val;
151  {
152  interrupts::Critical_section cs; // ----- Critical section -----
153 
154  // Sample ticks counter inside critical section.
155  ticks = __systick_now;
156 
157  // Initial sample of the decrementing counter.
158  // Might happen before the event, will be used as such.
159  val = SysTick->VAL;
160 
161  // Check overflow. If the exception is pending, it means the
162  // ticks counter was not yet updated, but the counter was reloaded.
163  if (SysTick->CTRL & SCB_ICSR_PENDSTSET_Msk)
164  {
165  // Sample the decrementing counter again to validate the
166  // initial sample.
167  uint32_t val_update = SysTick->VAL;
168  if (val_update > val)
169  {
170  // The second sample is more accurate.
171  val = val_update;
172  }
173  ticks++; // Adjust to next tick.
174  }
175  }
176  details->cycles = load - val;
177  details->ticks = ticks;
178 
179  return ticks;
180  }

212  {
213  os_assert_err(!scheduler::in_handler_mode (), EPERM);
214 
215  trace::printf ("%s(%d_ticks)\n", __func__, ticks);
216

218  Systick_clock::sleep_rep ticks_to_go = ticks;
219  for (;;)
220  {
221  result_t res;
222

224 
225  Systick_clock::rep now = Systick_clock::now ();
226  Systick_clock::sleep_rep slept_ticks = (Systick_clock::sleep_rep) (now
227  - prev);
228  if (slept_ticks >= ticks_to_go)
229  {
230  return ETIMEDOUT;
231  }
232 
233  if (this_thread::thread ().interrupted ())
234  {
235  return EINTR;
236  }
237 
238  if (res != ETIMEDOUT)
239  {
240  return res;
241  }
242 
243  prev = now;
244  ticks_to_go -= slept_ticks;
245  }
246  return ENOTRECOVERABLE;
