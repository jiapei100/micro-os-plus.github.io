395  result_t
396  semaphore::wait ()
397  {
398  os_assert_err(!interrupts::in_handler_mode (), EPERM);
399  os_assert_err(!scheduler::locked (), EPERM);
400 
401 #if defined(OS_TRACE_RTOS_SEMAPHORE)
402  trace::printf ("%s() @%p %s\n", __func__, this, name ());
403 #endif
404 
405 #if defined(OS_USE_RTOS_PORT_SEMAPHORE)
406 
407  return port::semaphore::wait (this);
408 
409 #else
410 
411  // Extra test before entering the loop, with its inherent weight.
412  // Trade size for speed.
413  {
414  interrupts::critical_section ics; // ----- Critical section -----
415 
416  if (_try_wait ())
417  {
418  return result::ok;
419  }
420  }
421 
422  thread& crt_thread = this_thread::thread ();
423 
424  // Prepare a list node pointing to the current thread.
425  // Do not worry for being on stack, it is temporarily linked to the
426  // list and guaranteed to be removed before this function returns.
427  waiting_thread_node node
428  { crt_thread };
429 
430  for (;;)
431  {
432  {
433  interrupts::critical_section ics; // ----- Critical section -----
434 
435  if (_try_wait ())
436  {
437  return result::ok;
438  }
439 
440  // Add this thread to the semaphore waiting list.
441  scheduler::_link_node (list_, node);
442  // state::suspended set in above link().
443  }
444 
445  port::scheduler::reschedule ();
446 
447  // Remove the thread from the semaphore waiting list,
448  // if not already removed by post().
449  scheduler::_unlink_node (node);
450 
451  if (crt_thread.interrupted ())
452  {
453  return EINTR;
454  }
455  }
456 
457  /* NOTREACHED */
458  return ENOTRECOVERABLE;
459 
460 #endif

569

576  thread& crt_thread = this_thread::thread ();
577 
578  // Prepare a list node pointing to the current thread.
579  // Do not worry for being on stack, it is temporarily linked to the
580  // list and guaranteed to be removed before this function returns.
581  waiting_thread_node node
582  { crt_thread };
583

585  clock::timestamp_t timeout_timestamp = clock_->steady_now () + timeout;
586 
587  // Prepare a timeout node pointing to the current thread.
588  timeout_thread_node timeout_node
589  { timeout_timestamp, crt_thread };
590 
591  for (;;)
592  {
593  {
594  interrupts::critical_section ics; // ----- Critical section -----
595 
596  if (_try_wait ())
597  {
598  return result::ok;
599  }
600 
601  // Add this thread to the semaphore waiting list,
602  // and the clock timeout list.
603  scheduler::_link_node (list_, node, clock_list, timeout_node);
604  // state::suspended set in above link().
605  }
606 
607  port::scheduler::reschedule ();
608 
609  // Remove the thread from the semaphore waiting list,
610  // if not already removed by post() and from the clock
611  // timeout list, if not already removed by the timer.
612  scheduler::_unlink_node (node, timeout_node);
613 
614  if (crt_thread.interrupted ())
615  {
616  return EINTR;
617  }
618 
619  if (clock_->steady_now () >= timeout_timestamp)
620  {
621  return ETIMEDOUT;
622  }
623  }
