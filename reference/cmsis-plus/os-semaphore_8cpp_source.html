419  semaphore::wait ()
420  {
421  os_assert_err(!interrupts::in_handler_mode (), EPERM);
422  os_assert_err(!scheduler::locked (), EPERM);
423 
424 #if defined(OS_TRACE_RTOS_SEMAPHORE)
425  trace::printf ("%s() @%p %s\n", __func__, this, name ());
426 #endif
427 
428 #if defined(OS_USE_RTOS_PORT_SEMAPHORE)
429 
430  return port::semaphore::wait (this);
431 
432 #else
433 
434  // Extra test before entering the loop, with its inherent weight.
435  // Trade size for speed.
436  {
437  interrupts::critical_section ics; // ----- Critical section -----
438 
439  if (_try_wait ())
440  {
441  return result::ok;
442  }
443  }
444 
445  thread& crt_thread = this_thread::thread ();
446 
447  // Prepare a list node pointing to the current thread.
448  // Do not worry for being on stack, it is temporarily linked to the
449  // list and guaranteed to be removed before this function returns.
450  waiting_thread_node node
451  { crt_thread };
452 
453  for (;;)
454  {
455  {
456  interrupts::critical_section ics; // ----- Critical section -----
457 
458  if (_try_wait ())
459  {
460  return result::ok;
461  }
462 
463  // Add this thread to the semaphore waiting list.
464  scheduler::_link_node (list_, node);
465  // state::suspended set in above link().
466  }
467 
468  port::scheduler::reschedule ();
469 
470  // Remove the thread from the semaphore waiting list,
471  // if not already removed by post().
472  scheduler::_unlink_node (node);
473 
474  if (crt_thread.interrupted ())
475  {
476  return EINTR;
477  }
478  }
479 
480  /* NOTREACHED */
481  return ENOTRECOVERABLE;
482 
483 #endif
