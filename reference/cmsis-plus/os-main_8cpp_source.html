75 int __attribute__((weak))
76 main (int argc, char* argv[])
77 {
78  using namespace os;
79  using namespace os::rtos;
80 
81  // At this stage the system clock should have already been configured
82  // at high speed.
83  trace::printf ("System clock: %u Hz\n", SystemCoreClock);
84 
85  scheduler::initialize ();
86 
87  // Since the native threads have a single argument, and we prefer to
88  // avoid C++11 tuples and function objects, there is no other simple
89  // way than to pack the 3 args in a structure and pass it via the
90  // trampoline to invoke the os_main().
91 
92  static thread::main_args_t args;
93  args.func = os_main;
94  args.argc = argc;
95  args.argv = argv;
96 
97  // Necessarily static, the initial stack will be used for the
98  // interrupts, and some implementations (like FreeRTOS) are not
99  // able to preserve this stack content.
100 
101  static stack::element_t main_stack[OS_INTEGER_RTOS_MAIN_STACK_SIZE_BYTES
102  / sizeof(stack::element_t)];
103 
104  static thread::Attributes attr
105  { "main" };
106  attr.th_stack_address = main_stack;
107  attr.th_stack_size_bytes = sizeof(main_stack);
108 
109  static Thread main_thread
110  { attr, (thread::func_t) thread::_main_trampoline,
111  (thread::func_args_t) &args };
112 
113  scheduler::start ();
114 
115  // Some (most?) embedded schedulers do not return after start(),
116  // but on POSIX synthetic platforms they do,
117  // so wait for the main thread to terminate.
118  main_thread.join ();
119 
120  return 0;
121 }
