608  message_queue::_init (void)

612 #if !defined(OS_USE_RTOS_PORT_MESSAGE_QUEUE)
613 
614  // Construct a linked list of blocks. Store the pointer at
615  // the beginning of each block. Each block
616  // will hold the address of the next free block,
617  // or `nullptr` at the end.
618  char* p = static_cast<char*> (queue_addr_);
619  for (std::size_t i = 1; i < msgs_; ++i)
620  {
621  // Compute the address of the next block;
622  char* pn = p + msg_size_bytes_;
623 
624  // Make this block point to the next one.
625  *(static_cast<void**> (static_cast<void*> (p))) = pn;
626  // Advance pointer
627  p = pn;
628  }
629 
630  // Mark end of list.
631  *(static_cast<void**> (static_cast<void*> (p))) = nullptr;
632 
633  first_free_ = queue_addr_; // Pointer to first block.
634 
635  head_ = no_index;
636 
637  if (!send_list_.empty ())
638  {
639  // Wake-up all threads, if any.
640  send_list_.resume_all ();
641 
642  send_list_.clear ();
643  }
644

648  receive_list_.resume_all ();

659  bool
660  message_queue::_try_send (const void* msg, std::size_t nbytes,
661  priority_t mprio)
662  {
663  if (first_free_ == nullptr)
664  {
665  // No available space to send the message.
666  return false;
667  }
668 
669  // Get the address where the message will be copied.
670  // This is the first free memory block.
671  char* dest = static_cast<char*> (first_free_);
672 
673  // Update to next free, if any (the last one has nullptr).
674  first_free_ = *(static_cast<void**> (first_free_));
675 
676  // Using the address, compute the index in the array.
677  std::size_t msg_ix = (static_cast<std::size_t> (dest
678  - static_cast<char*> (queue_addr_)) / msg_size_bytes_);
679  prio_array_[msg_ix] = mprio;
680 
681  if (head_ == no_index)
682  {
683  // No other message in the queue, enlist this one
684  // as head, with links to itself.
685  head_ = static_cast<index_t> (msg_ix);
686  prev_array_[msg_ix] = static_cast<index_t> (msg_ix);
687  next_array_[msg_ix] = static_cast<index_t> (msg_ix);
688  }
689  else
690  {
691  std::size_t ix;
692  // Arrange to insert between head and tail.
693  ix = prev_array_[head_];
694  // Check if the priority is higher than the head priority.
695  if (mprio > prio_array_[head_])
696  {
697  // Having the highest priority, the new message
698  // becomes the new head.
699  head_ = static_cast<index_t> (msg_ix);
700  }
701  else
702  {
703  // If not higher than the head, try to insert at the tail,
704  // but advance up until the same priority is found.
705  while ((mprio > prio_array_[ix]))
706  {
707  ix = prev_array_[ix];
708  }
709  }
710  prev_array_[msg_ix] = static_cast<index_t> (ix);
711  next_array_[msg_ix] = next_array_[ix];
712 
713  // Break the chain and insert the new index.
714  std::size_t tmp_ix = next_array_[ix];
715  next_array_[ix] = static_cast<index_t> (msg_ix);
716  prev_array_[tmp_ix] = static_cast<index_t> (msg_ix);
717  }
718 
719  // One more message added to the queue.
720  ++count_;
721 
722  {
723  // ----- Enter uncritical section -----------------------------------
724  interrupts::uncritical_section iucs;
725 
726  // Copy message from user buffer to queue storage.
727  std::memcpy (dest, msg, nbytes);
728  if (nbytes < msg_size_bytes_)
729  {
730  // Fill in the remaining space with 0x00.
731  std::memset (dest + nbytes, 0x00, msg_size_bytes_ - nbytes);
732  }
733 
734  // Wake-up one thread, if any.
735  receive_list_.resume_one ();
736 
737  // ----- Exit uncritical section ------------------------------------
738  }
739  return true;
740  }

746  bool
747  message_queue::_try_receive (void* msg, std::size_t nbytes,
748  priority_t* mprio)
749  {
750 
751  if (head_ == no_index)
752  {
753  return false;
754  }
755 
756  char* src = static_cast<char*> (queue_addr_) + head_ * msg_size_bytes_;
757 #if defined(OS_TRACE_RTOS_MQUEUE_)
758  trace::printf ("%s(%p,%u) @%p %s src %p %p\n", __func__, msg, nbytes,
759  this, name (), src, first_free_);
760 #endif
761 
762  {
763  // ----- Enter uncritical section -----
764  interrupts::uncritical_section iucs;
765 
766  // Copy message from queue to user buffer.
767  memcpy (msg, src, nbytes);
768  if (mprio != nullptr)
769  {
770  *mprio = prio_array_[head_];
771  }
772 
773  // ----- Exit uncritical section -----
774  }
775 
776  {
777  if (count_ > 1)
778  {
779  // Remove the current element from the list.
780  prev_array_[next_array_[head_]] = prev_array_[head_];
781  next_array_[prev_array_[head_]] = next_array_[head_];
782 
783  // Next becomes the new head.
784  head_ = next_array_[head_];
785  }
786  else
787  {
788  // If there was only one, the list is empty now.
789  head_ = no_index;
790  }
791 
792  // Perform a push_front() on the single linked LIFO list,
793  // i.e. add the block to the beginning of the list.
794 
795  // Link previous list to this block; may be null, but it does
796  // not matter.
797  *(static_cast<void**> (static_cast<void*> (src))) = first_free_;
798 
799  // Now this block is the first one.
800  first_free_ = src;
801 
802  --count_;
803  }
804 
805  // Wake-up one thread, if any.
806  send_list_.resume_one ();
807 
808  return true;
809  }

860  os_assert_err(msg != nullptr, EINVAL);
861  os_assert_err(nbytes <= msg_size_bytes_, EMSGSIZE);
862 
863 #if defined(OS_TRACE_RTOS_MQUEUE)
864  trace::printf ("%s(%p,%d,%d) @%p %s\n", __func__, msg, nbytes, mprio,
865  this, name ());

869

871 
872 #else
873 
874  {
875  interrupts::critical_section ics; // ----- Critical section -----
876 
877  if (_try_send (msg, nbytes, mprio))
878  {
879  return result::ok;
880  }
881  }
882 
883  thread& crt_thread = this_thread::thread ();
884 
885  // Prepare a list node pointing to the current thread.
886  // Do not worry for being on stack, it is temporarily linked to the
887  // list and guaranteed to be removed before this function returns.
888  waiting_thread_node node
889  { crt_thread };

893  {

907 
908  // Remove the thread from the message queue send waiting list,
909  // if not already removed by receive().
910  scheduler::_unlink_node (node);
911 
912  if (crt_thread.interrupted ())
913  {
914  return EINTR;
915  }
916  }

919  return ENOTRECOVERABLE;
920 
921 #endif
