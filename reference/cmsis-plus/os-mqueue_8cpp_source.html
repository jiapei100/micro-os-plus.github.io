340  bool
341  Message_queue::_try_send (const char* msg, std::size_t nbytes,
342  mqueue::priority_t mprio)
343  {
344  char* dest;
345 
346  {
347  interrupts::Critical_section cs; // ----- Critical section -----
348 
349  if (first_free_ == nullptr)
350  {
351  // No available space to send the message.
352  return false;
353  }
354 
355  // Get the address where the message will be copied.
356  // This is the first free memory block.
357  dest = (char*) first_free_;
358 
359  // Update to next free, if any (the last one has nullptr).
360  first_free_ = *(void**) first_free_;
361 
362  // Using the address, compute the index in the array.
363  std::size_t msg_ix = (dest - (char*) queue_addr_) / msg_size_bytes_;
364  prio_array_[msg_ix] = mprio;
365 
366  if (head_ == mqueue::no_index)
367  {
368  // No other message in the queue, enlist this one
369  // as head, with links to itself.
370  head_ = (mqueue::index_t) msg_ix;
371  prev_array_[msg_ix] = (mqueue::index_t) msg_ix;
372  next_array_[msg_ix] = (mqueue::index_t) msg_ix;
373  }
374  else
375  {
376  std::size_t ix;
377  // Arrange to insert between head and tail.
378  ix = prev_array_[head_];
379  // Check if the priority is higher than the head priority.
380  if (mprio > prio_array_[head_])
381  {
382  // Having the highest priority, the new message
383  // becomes the new head.
384  head_ = (mqueue::index_t) msg_ix;
385  }
386  else
387  {
388  // If not higher than the head, try to insert at the tail,
389  // but advance up until the same priority is found.
390  while ((mprio > prio_array_[ix]))
391  {
392  ix = prev_array_[ix];
393  }
394  }
395  prev_array_[msg_ix] = (mqueue::index_t) ix;
396  next_array_[msg_ix] = next_array_[ix];
397 
398  // Break the chain and insert the new index.
399  std::size_t tmp_ix = next_array_[ix];
400  next_array_[ix] = (mqueue::index_t) msg_ix;
401  prev_array_[tmp_ix] = (mqueue::index_t) msg_ix;
402  }
403 
404  // One more message added to the queue.
405  ++count_;
406  }
407 
408  // Copy message from user buffer to queue storage.
409  std::memcpy (dest, msg, nbytes);
410  if (nbytes < msg_size_bytes_)
411  {
412  // Fill in the remaining space with 0x00.
413  std::memset (dest + nbytes, 0x00, msg_size_bytes_ - nbytes);
414  }
415 
416  // Wake-up one thread, if any.
417  receive_list_.wakeup_one ();
418 
419  return true;
420  }

422 #endif /* !defined(OS_INCLUDE_RTOS_PORT_MESSAGE_QUEUE) */

691  bool
692  Message_queue::_try_receive (char* msg, std::size_t nbytes,
693  mqueue::priority_t* mprio)
694  {
695  char* src;
696 
697  {
698  interrupts::Critical_section cs; // ----- Critical section -----
699 
700  if (head_ == mqueue::no_index)
701  {
702  return false;
703  }
704 
705  src = (char*) queue_addr_ + head_ * msg_size_bytes_;
706  }
707 
708  // Copy message from queue to user buffer.
709  memcpy (msg, src, nbytes);
710  if (mprio != nullptr)
711  {
712  *mprio = prio_array_[head_];
713  }
714 
715  {
716  interrupts::Critical_section cs; // ----- Critical section -----
717 
718  if (count_ > 1)
719  {
720  // Remove the current element from the list.
721  prev_array_[next_array_[head_]] = prev_array_[head_];
722  next_array_[prev_array_[head_]] = next_array_[head_];
723 
724  // Next becomes the new head.
725  head_ = next_array_[head_];
726  }
727  else
728  {
729  // If there was only one, the list is empty now.
730  head_ = mqueue::no_index;
731  }
732 
733  // Perform a push_front() on the single linked LIFO list,
734  // i.e. add the block to the beginning of the list.
735 
736  // Link previous list to this block; may be null, but it does
737  // not matter.
738  *(void**) src = first_free_;
739 
740  // Now this block is the first one.
741  first_free_ = src;
742 
743  --count_;
744  }
745 
746  // Wake-up one thread, if any.
747  send_list_.wakeup_one ();
748 
749  return true;
750  }
