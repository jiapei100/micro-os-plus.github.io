337  bool
338  Message_queue::_try_send (const char* msg, std::size_t nbytes,
339  mqueue::priority_t mprio)
340  {
341  if (first_free_ == nullptr)
342  {
343  // No available space to send the message.
344  return false;
345  }
346 
347  // Get the address where the message will be copied.
348  // This is the first free memory block.
349  char* p = (char*) first_free_;
350 
351  // Update to next free, if any (the last one has nullptr).
352  first_free_ = *(void**) first_free_;
353 
354  // Copy message to queue storage.
355  std::memcpy (p, msg, nbytes);
356  if (nbytes < msg_size_bytes_)
357  {
358  // Fill in the remaining space with 0x00.
359  std::memset (p + nbytes, 0x00, msg_size_bytes_ - nbytes);
360  }
361 
362  // Using the address, compute the index in the array.
363  std::size_t msg_ix = (p - (char*) queue_addr_) / msg_size_bytes_;
364  prio_array_[msg_ix] = mprio;
365 
366  if (head_ == mqueue::no_index)
367  {
368  // No other message in the queue, enlist this one
369  // as head, with links to itself.
370  head_ = (mqueue::index_t) msg_ix;
371  prev_array_[msg_ix] = (mqueue::index_t) msg_ix;
372  next_array_[msg_ix] = (mqueue::index_t) msg_ix;
373  }
374  else
375  {
376  std::size_t ix;
377  // Arrange to insert between head and tail.
378  ix = prev_array_[head_];
379  // Check if the priority is higher than the head priority.
380  if (mprio > prio_array_[head_])
381  {
382  // Having the highest priority, the new message
383  // becomes the new head.
384  head_ = (mqueue::index_t) msg_ix;
385  }
386  else
387  {
388  // If not higher than the head, try to insert at the tail,
389  // but advance up until the same priority is found.
390  while ((mprio > prio_array_[ix]))
391  {
392  ix = prev_array_[ix];
393  }
394  }
395  prev_array_[msg_ix] = (mqueue::index_t) ix;
396  next_array_[msg_ix] = next_array_[ix];
397 
398  // Break the chain and insert the new index.
399  std::size_t tmp_ix = next_array_[ix];
400  next_array_[ix] = (mqueue::index_t) msg_ix;
401  prev_array_[tmp_ix] = (mqueue::index_t) msg_ix;
402  }
403 
404  // One more message added to the queue.
405  ++count_;
406 
407  if (!receive_list_.empty ())
408  {
409  // Wake-up one thread, if any.
410  receive_list_.wakeup_one ();
411  }
412 
413  return true;
414  }

674  bool
675  Message_queue::_try_receive (char* msg, std::size_t nbytes,
676  mqueue::priority_t* mprio)
677  {
678  if (head_ == mqueue::no_index)
679  {
680  return false;
681  }
682 
683  char* p = (char*) queue_addr_ + head_ * msg_size_bytes_;
684  memcpy (msg, p, nbytes);
685  if (mprio != nullptr)
686  {
687  *mprio = prio_array_[head_];
688  }
689 
690  if (count_ > 1)
691  {
692  // Remove the current element from the list.
693  prev_array_[next_array_[head_]] = prev_array_[head_];
694  next_array_[prev_array_[head_]] = next_array_[head_];
695 
696  // Next becomes the new head.
697  head_ = next_array_[head_];
698  }
699  else
700  {
701  // If there was only one, the list is empty now.
702  head_ = mqueue::no_index;
703  }
704 
705  // Perform a push_front() on the single linked LIFO list,
706  // i.e. add the block to the beginning of the list.
707 
708  // Link previous list to this block; may be null, but it does
709  // not matter.
710  *(void**) p = first_free_;
711 
712  // Now this block is the first one.
713  first_free_ = p;
714 
715  --count_;
716 
717  if (!send_list_.empty ())
718  {
719  // Wake-up one thread, if any.
720  send_list_.wakeup_one ();
721  }
722 
723  return true;
724  }
