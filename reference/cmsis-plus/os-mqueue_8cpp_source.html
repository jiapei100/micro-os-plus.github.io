355  bool
356  Message_queue::_try_send (const char* msg, std::size_t nbytes,
357  mqueue::priority_t mprio)
358  {
359  char* dest;
360 
361  if (first_free_ == nullptr)
362  {
363  // No available space to send the message.
364  return false;
365  }
366 
367  // Get the address where the message will be copied.
368  // This is the first free memory block.
369  dest = (char*) first_free_;
370 
371  // Update to next free, if any (the last one has nullptr).
372  first_free_ = *(void**) first_free_;
373 
374  // Using the address, compute the index in the array.
375  std::size_t msg_ix = (dest - (char*) queue_addr_) / msg_size_bytes_;
376  prio_array_[msg_ix] = mprio;
377 
378  if (head_ == mqueue::no_index)
379  {
380  // No other message in the queue, enlist this one
381  // as head, with links to itself.
382  head_ = (mqueue::index_t) msg_ix;
383  prev_array_[msg_ix] = (mqueue::index_t) msg_ix;
384  next_array_[msg_ix] = (mqueue::index_t) msg_ix;
385  }
386  else
387  {
388  std::size_t ix;
389  // Arrange to insert between head and tail.
390  ix = prev_array_[head_];
391  // Check if the priority is higher than the head priority.
392  if (mprio > prio_array_[head_])
393  {
394  // Having the highest priority, the new message
395  // becomes the new head.
396  head_ = (mqueue::index_t) msg_ix;
397  }
398  else
399  {
400  // If not higher than the head, try to insert at the tail,
401  // but advance up until the same priority is found.
402  while ((mprio > prio_array_[ix]))
403  {
404  ix = prev_array_[ix];
405  }
406  }
407  prev_array_[msg_ix] = (mqueue::index_t) ix;
408  next_array_[msg_ix] = next_array_[ix];
409 
410  // Break the chain and insert the new index.
411  std::size_t tmp_ix = next_array_[ix];
412  next_array_[ix] = (mqueue::index_t) msg_ix;
413  prev_array_[tmp_ix] = (mqueue::index_t) msg_ix;
414  }
415 
416  // One more message added to the queue.
417  ++count_;
418 
419  {
420  // ----- Enter uncritical section -----------------------------------
421  interrupts::Uncritical_section iucs;
422 
423  // Copy message from user buffer to queue storage.
424  std::memcpy (dest, msg, nbytes);
425  if (nbytes < msg_size_bytes_)
426  {
427  // Fill in the remaining space with 0x00.
428  std::memset (dest + nbytes, 0x00, msg_size_bytes_ - nbytes);
429  }
430 
431  // Wake-up one thread, if any.
432  receive_list_.resume_one ();
433  // ----- Exit uncritical section ------------------------------------
434  }
435  return true;
436  }

744  bool
745  Message_queue::_try_receive (char* msg, std::size_t nbytes,
746  mqueue::priority_t* mprio)
747  {
748  char* src;
749 
750  {
751  if (head_ == mqueue::no_index)
752  {
753  return false;
754  }
755 
756  src = (char*) queue_addr_ + head_ * msg_size_bytes_;
757  }
758 
759  {
760  // ----- Enter uncritical section -----
761  interrupts::Uncritical_section iucs;
762 
763  // Copy message from queue to user buffer.
764  memcpy (msg, src, nbytes);
765  if (mprio != nullptr)
766  {
767  *mprio = prio_array_[head_];
768  }
769 
770  // ----- Exit uncritical section -----
771  }
772 
773  {
774  if (count_ > 1)
775  {
776  // Remove the current element from the list.
777  prev_array_[next_array_[head_]] = prev_array_[head_];
778  next_array_[prev_array_[head_]] = next_array_[head_];
779 
780  // Next becomes the new head.
781  head_ = next_array_[head_];
782  }
783  else
784  {
785  // If there was only one, the list is empty now.
786  head_ = mqueue::no_index;
787  }
788 
789  // Perform a push_front() on the single linked LIFO list,
790  // i.e. add the block to the beginning of the list.
791 
792  // Link previous list to this block; may be null, but it does
793  // not matter.
794  *(void**) src = first_free_;
795 
796  // Now this block is the first one.
797  first_free_ = src;
798 
799  --count_;
800  }
801 
802  // Wake-up one thread, if any.
803  send_list_.resume_one ();
804 
805  return true;
806  }
