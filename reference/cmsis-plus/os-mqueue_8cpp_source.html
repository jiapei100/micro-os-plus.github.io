493  // Construct a linked list of blocks. Store the pointer at
494  // the beginning of each block. Each block
495  // will hold the address of the next free block,
496  // or `nullptr` at the end.
497  char* p = static_cast<char*> (queue_addr_);
498  for (std::size_t i = 1; i < msgs_; ++i)
499  {
500  // Compute the address of the next block;
501  char* pn = p + msg_size_bytes_;
502 
503  // Make this block point to the next one.
504  *(static_cast<void**> (static_cast<void*> (p))) = pn;
505  // Advance pointer
506  p = pn;
507  }
508 
509  // Mark end of list.
510  *(static_cast<void**> (static_cast<void*> (p))) = nullptr;
511 
512  first_free_ = queue_addr_; // Pointer to first block.
513 
514  head_ = mqueue::no_index;
515 
516  if (!send_list_.empty ())
517  {
518  // Wake-up all threads, if any.
519  send_list_.resume_all ();
520 
521  send_list_.clear ();
522  }
523 
524  if (!receive_list_.empty ())
525  {
526  // Wake-up all threads, if any.
527  receive_list_.resume_all ();
528 
529  receive_list_.clear ();

538  bool
539  Message_queue_base::_try_send (const void* msg, std::size_t nbytes,
540  mqueue::priority_t mprio)
541  {
542  if (first_free_ == nullptr)
543  {
544  // No available space to send the message.
545  return false;
546  }
547 
548  // Get the address where the message will be copied.
549  // This is the first free memory block.
550  char* dest = static_cast<char*> (first_free_);
551 
552  // Update to next free, if any (the last one has nullptr).
553  first_free_ = *(static_cast<void**> (first_free_));
554 
555  // Using the address, compute the index in the array.
556  std::size_t msg_ix = (static_cast<std::size_t> (dest
557  - static_cast<char*> (queue_addr_)) / msg_size_bytes_);
558  prio_array_[msg_ix] = mprio;
559 
560  if (head_ == mqueue::no_index)
561  {
562  // No other message in the queue, enlist this one
563  // as head, with links to itself.
564  head_ = static_cast<mqueue::index_t> (msg_ix);
565  prev_array_[msg_ix] = static_cast<mqueue::index_t> (msg_ix);
566  next_array_[msg_ix] = static_cast<mqueue::index_t> (msg_ix);
567  }
568  else
569  {
570  std::size_t ix;
571  // Arrange to insert between head and tail.
572  ix = prev_array_[head_];
573  // Check if the priority is higher than the head priority.
574  if (mprio > prio_array_[head_])
575  {
576  // Having the highest priority, the new message
577  // becomes the new head.
578  head_ = static_cast<mqueue::index_t> (msg_ix);
579  }
580  else
581  {
582  // If not higher than the head, try to insert at the tail,
583  // but advance up until the same priority is found.
584  while ((mprio > prio_array_[ix]))
585  {
586  ix = prev_array_[ix];
587  }
588  }
589  prev_array_[msg_ix] = static_cast<mqueue::index_t> (ix);
590  next_array_[msg_ix] = next_array_[ix];
591 
592  // Break the chain and insert the new index.
593  std::size_t tmp_ix = next_array_[ix];
594  next_array_[ix] = static_cast<mqueue::index_t> (msg_ix);
595  prev_array_[tmp_ix] = static_cast<mqueue::index_t> (msg_ix);
596  }
597 
598  // One more message added to the queue.
599  ++count_;
600 
601  {
602  // ----- Enter uncritical section -----------------------------------
603  interrupts::Uncritical_section iucs;
604 
605  // Copy message from user buffer to queue storage.
606  std::memcpy (dest, msg, nbytes);
607  if (nbytes < msg_size_bytes_)
608  {
609  // Fill in the remaining space with 0x00.
610  std::memset (dest + nbytes, 0x00, msg_size_bytes_ - nbytes);
611  }
612 
613  // Wake-up one thread, if any.
614  receive_list_.resume_one ();
615 
616  // ----- Exit uncritical section ------------------------------------
617  }
618  return true;
619  }

933  bool
934  Message_queue_base::_try_receive (void* msg, std::size_t nbytes,
935  mqueue::priority_t* mprio)
936  {
937 
938  if (head_ == mqueue::no_index)
939  {
940  return false;
941  }
942 
943  char* src = static_cast<char*> (queue_addr_) + head_ * msg_size_bytes_;
944 #if defined(OS_TRACE_RTOS_MQUEUE_)
945  trace::printf ("%s(%p,%d) @%p %s src %p %p\n", __func__, msg, nbytes,
946  this, name (), src, first_free_);
947 #endif
948 
949  {
950  // ----- Enter uncritical section -----
951  interrupts::Uncritical_section iucs;
952 
953  // Copy message from queue to user buffer.
954  memcpy (msg, src, nbytes);
955  if (mprio != nullptr)
956  {
957  *mprio = prio_array_[head_];
958  }
959 
960  // ----- Exit uncritical section -----
961  }
962 
963  {
964  if (count_ > 1)
965  {
966  // Remove the current element from the list.
967  prev_array_[next_array_[head_]] = prev_array_[head_];
968  next_array_[prev_array_[head_]] = next_array_[head_];
969 
970  // Next becomes the new head.
971  head_ = next_array_[head_];
972  }
973  else
974  {
975  // If there was only one, the list is empty now.
976  head_ = mqueue::no_index;
977  }
978 
979  // Perform a push_front() on the single linked LIFO list,
980  // i.e. add the block to the beginning of the list.
981 
982  // Link previous list to this block; may be null, but it does
983  // not matter.
984  *(static_cast<void**> (static_cast<void*> (src))) = first_free_;
985 
986  // Now this block is the first one.
987  first_free_ = src;
988 
989  --count_;
990  }
991 
992  // Wake-up one thread, if any.
993  send_list_.resume_one ();
994 
995  return true;
996  }
